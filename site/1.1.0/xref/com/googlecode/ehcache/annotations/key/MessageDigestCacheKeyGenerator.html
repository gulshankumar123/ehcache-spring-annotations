83      @Override

85          final MessageDigest messageDigest = generator.getMessageDigest();

93      protected String encodeHash(final byte[] digest) {
94          return Base64.encodeBase64URLSafeString(digest);
95      }

97      /**
98       * Tries to clone the {@link MessageDigest} that was created during construction. If the clone fails
99       * that is remembered and from that point on new {@link MessageDigest} instances will be created on
100      * every call.
101      * 
102      * @return Generates a {@link MessageDigest} to use during a call to {@link #generateKey(Object...)}
103      */
104     protected MessageDigest getMessageDigest() {
105         if (this.cloneNotSupported) {
106             final String algorithm = this.baseMessageDigest.getAlgorithm();
107             try {
108                 return MessageDigest.getInstance(algorithm);
109             }
110             catch (NoSuchAlgorithmException e) {
111                 throw new IllegalStateException("MessageDigest algorithm '" + algorithm + "' was supported when " + this.getClass().getSimpleName() + " was created but is not now. This should not be possible.", e);
112             }
113         }
114         
115         try {
116             return (MessageDigest)this.baseMessageDigest.clone();
117         }
118         catch (CloneNotSupportedException e) {
119             this.cloneNotSupported = true;
120             this.logger.warn("Could not clone MessageDigest using algorithm '" + this.baseMessageDigest.getAlgorithm() + "'. MessageDigest.getInstance will be used from now on which will be much more expensive.", e);
121             return this.getMessageDigest();
122         }
123     }

195         else if (e instanceof Boolean) {

204         else if (e instanceof Double) {

206         }
207         else if (e instanceof Float) {
208             generator.writeFloat((Float)e);
209         }
210         else if (e instanceof Integer) {
211             generator.writeInt((Integer)e);
212         }
213         else if (e instanceof Long) {
214             generator.writeLong((Long)e);
215         }
216         else if (e instanceof Short) {
217             generator.writeShort((Short)e);
218         }
