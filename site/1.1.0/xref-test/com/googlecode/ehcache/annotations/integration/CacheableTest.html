93      @Test
94      public void testExceptionCaching() {
95          Assert.assertEquals(0, cacheableTestInterface.interfaceAnnotatedExceptionCachedCount());
96          Assert.assertEquals(0, cacheableTestInterface.interfaceAnnotatedExceptionCachedThrowsCount());
97          
98          Assert.assertEquals("interfaceAnnotatedExceptionCached(false)", cacheableTestInterface.interfaceAnnotatedExceptionCached(false));
99          Assert.assertEquals(1, cacheableTestInterface.interfaceAnnotatedExceptionCachedCount());
100         Assert.assertEquals(0, cacheableTestInterface.interfaceAnnotatedExceptionCachedThrowsCount());
101         
102         Assert.assertEquals("interfaceAnnotatedExceptionCached(false)", cacheableTestInterface.interfaceAnnotatedExceptionCached(false));
103         Assert.assertEquals(1, cacheableTestInterface.interfaceAnnotatedExceptionCachedCount());
104         Assert.assertEquals(0, cacheableTestInterface.interfaceAnnotatedExceptionCachedThrowsCount());
105         
106         try {
107             cacheableTestInterface.interfaceAnnotatedExceptionCached(true);
108             Assert.fail("interfaceAnnotatedExceptionCached(true) should have thrown an exception");
109         }
110         catch (RuntimeException re) {
111             Assert.assertEquals("throwsException was true", re.getMessage());
112         }
113         Assert.assertEquals(1, cacheableTestInterface.interfaceAnnotatedExceptionCachedCount());
114         Assert.assertEquals(1, cacheableTestInterface.interfaceAnnotatedExceptionCachedThrowsCount());
115         
116         try {
117             cacheableTestInterface.interfaceAnnotatedExceptionCached(true);
