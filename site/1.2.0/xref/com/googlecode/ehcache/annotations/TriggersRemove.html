16  package com.googlecode.ehcache.annotations;
17  
18  import java.lang.annotation.Documented;
19  import java.lang.annotation.ElementType;
20  import java.lang.annotation.Inherited;
21  import java.lang.annotation.Retention;
22  import java.lang.annotation.RetentionPolicy;
23  import java.lang.annotation.Target;
24  
25  import com.googlecode.ehcache.annotations.key.CacheKeyGenerator;
26  import com.googlecode.ehcache.annotations.resolver.CacheResolverFactory;
27  
28  
29  /**
30   * Annotation for methods who's calls trigger removal of data from a cache
31   */
32  @Target( { ElementType.METHOD })
33  @Retention(RetentionPolicy.RUNTIME)
34  @Inherited
35  @Documented
36  public @interface TriggersRemove {
37      /**
38       * The name(s) of the cache to use. Note that the same {@link CacheKeyGenerator} is used for all caches.
39       */
40      String[] cacheName();
41  
42      /**
43       * If a call should remove all elements from the cache. Will result in the specified keyGeneratorName 
44       * being ignored.
45       */
46      boolean removeAll() default false;
47      
48      /**
49       * The Spring Bean name of the {@link CacheKeyGenerator} to use for all caches.
50       * Ignored if {@link #keyGenerator()} is specified. 
51       * If this and {@link #keyGenerator()} are not specified the default generator will be used.
52       * If {@link #removeAll()} is true no {@link CacheKeyGenerator} is used.
53       */
54      String keyGeneratorName() default "";
55      
56      /**
57       * Used the specify and configure the {@link CacheKeyGenerator} to use for all caches.
58       * If this and {@link #keyGeneratorName()} are not specified the default generator will be used.  
59       * If {@link #removeAll()} is true no {@link CacheKeyGenerator} is used.
60       */
61      KeyGenerator keyGenerator() default @KeyGenerator(name = "");
62      
63      /**
64       * Used to specify 'when' to run the ehcache remove call:
65       * before or after method invocation.
66       * If you specify {@link When#AFTER_METHOD_INVOCATION} and your method invocation throws
67       * an exception, the ehcache remove call WILL NOT execute.
68       * Default value is {@link When#BEFORE_METHOD_INVOCATION}.
69       */
70      When when() default When.BEFORE_METHOD_INVOCATION;
71      
72      /**
73       * The Spring Bean name of the {@link TriggersRemoveInterceptor} to use when intercepting invocations.
74       */
75      String triggersRemoveInteceptorName() default "";
76      
77      /**
78       * The Spring Bean name of the {@link CacheResolverFactory} to use to resolve the correct cache at runtime.
79       * Ignored if {@link #resolverFactory()} is specified.
80       * If this and {@link #resolverFactory()} are not specified the specified {@link #cacheName()} will be used.
81       */
82      String resolverFactoryName() default "";
83      
84      /**
85       * Used the specify and configure the {@link CacheResolverFactory} to use.
86       * If this and {@link #resolverFactoryName()} are not specified the specified {@link #cacheName()} will be used. 
87       */
88      ResolverFactory resolverFactory() default @ResolverFactory(name = "");
89  }
