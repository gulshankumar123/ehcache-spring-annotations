24  import net.sf.ehcache.constructs.blocking.SelfPopulatingCache;

26  import org.junit.Assert;
27  import org.junit.Test;
28  import org.slf4j.Logger;
29  import org.slf4j.LoggerFactory;

31  import com.googlecode.ehcache.annotations.util.ThreadGroupRunner;

34   * @author Eric Dalquist

37  public class SelfPopulatingCacheTest {
38      protected final Logger logger = LoggerFactory.getLogger(this.getClass());
39      
40      private final AtomicInteger invocationCount = new AtomicInteger(0);
41      
42      @Test
43      public void testSelfPopulatingCache() throws Exception {
44          final int threads = 10;
45          
46          final CacheManager cacheManager = new CacheManager();
47          
48          cacheManager.addCache("testCache");
49          final Ehcache ehcache = cacheManager.getEhcache("testCache");
50          
51          final CountDownLatch threadRunningLatch = new CountDownLatch(threads + 1);
52          final CountDownLatch proceedLatch = new CountDownLatch(1);
53          invocationCount.set(0);
54          
55          final SelfPopulatingCache selfPopulatingCache = new SelfPopulatingCache(ehcache, new CacheEntryFactory() {
56              public Object createEntry(Object key) throws Exception {
57                  logger.trace("Enter createEntry({})", key);
58                  threadRunningLatch.countDown();
59                  try {
60                      logger.trace("Waiting in createEntry({})", key);
61                      proceedLatch.await();
62                  }
63                  catch (InterruptedException e) {
64                      throw new RuntimeException(e);
65                  }
66                  invocationCount.incrementAndGet();
67                  logger.trace("Returning from createEntry({})", key);
68                  return "Created: " + key;
69              }
70          });
71          
72          final ThreadGroupRunner threadGroup = new ThreadGroupRunner("SelfPopulatingTest-", true);
73          
74          threadGroup.addTask(threads, new Runnable() { 
75              public void run() {
76                  threadRunningLatch.countDown();
77                  logger.trace("Calling get(test1)");
78                  selfPopulatingCache.get("test1");
79              }
80          });
81          
82          // start the threads
83          logger.trace("Starting threads");
84          threadGroup.start();
85          
86          // wait for both threads to get going
87          logger.trace("Waiting for threads to start");
88          threadRunningLatch.await();
89          
90          // Let both threads complete
91          logger.trace("Signal threads to proceed");
92          proceedLatch.countDown();
93          
94          logger.trace("Waiting for threads to complete");
95          threadGroup.join();
96          
97          // verify only 1 call
98          Assert.assertEquals(1, invocationCount.get());
99      }
