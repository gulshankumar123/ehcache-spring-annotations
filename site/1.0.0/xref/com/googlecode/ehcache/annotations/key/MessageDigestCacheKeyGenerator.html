28  /**
29   * This key generator is a good option when a secure hash of the arguments is needed or when a
30   * larger final key space is desired. As with {@link HashCodeCacheKeyGenerator} the argument
31   * hashCode is used for key generation data but instead of the simple Java hash algorithm being
32   * used any {@link MessageDigest} can be used, SHA-1 is used by default. The generated hash is
33   * base64 encoded using {@link Base64#encodeBase64URLSafeString(byte[])}
34   * 
35   * <table>
36   *  <tr>
37   *      <th>Pros</th>
38   *      <th>Cons</th>
39   *  </tr>
40   *  <tr>
41   *      <td>
42   *          Better assurance than {@link HashCodeCacheKeyGenerator} against key collisions though
43   *          with the argument hashCodes being the source of the data it is still not perfect.
44   *      </td>
45   *      <td>
46   *          Slower than {@link HashCodeCacheKeyGenerator}
47   *      </td>
48   *  </tr>
49   * </table>
50   * 
51   * @author Eric Dalquist
52   * @version $Revision$
53   */
54  public class MessageDigestCacheKeyGenerator extends AbstractCacheKeyGenerator<String> {
55      public static final String DEFAULT_BEAN_NAME = "com.googlecode.ehcache.annotations.key.MessageDigestCacheKeyGenerator.DEFAULT_BEAN_NAME";
56      public static final String DEFAULT_ALGORITHM = "SHA-1";
57      public static final int HASH_CODE_BYTE_SIZE = 4;
58      
59      private static final byte[] ZERO_AS_BYTES = new byte[] {0, 0, 0, 0};
60      
61      protected final Logger logger = LoggerFactory.getLogger(this.getClass());
62      
63      private final MessageDigest messageDigest;
64      private boolean cloneNotSupported;
65      
66      public MessageDigestCacheKeyGenerator() throws NoSuchAlgorithmException {
67          this(DEFAULT_ALGORITHM, true, false);
68      }
69      
70      public MessageDigestCacheKeyGenerator(String algorithm) throws NoSuchAlgorithmException {
71          this(algorithm, true, false);
72      }
73      
74      public MessageDigestCacheKeyGenerator(boolean includeMethod, boolean includeParameterTypes) throws NoSuchAlgorithmException {
75          this(DEFAULT_ALGORITHM, includeMethod, includeParameterTypes);
76      }
77      
78      public MessageDigestCacheKeyGenerator(String algorithm, boolean includeMethod, boolean includeParameterTypes) throws NoSuchAlgorithmException {
79          super(includeMethod, includeParameterTypes);
80          this.messageDigest = MessageDigest.getInstance(algorithm);
81      }
82      
83  
84      @Override
85      protected String generateKey(Object... data) {
86          MessageDigest digester;
87          try {
88              digester = (MessageDigest) this.messageDigest.clone();
89          }
90          catch (CloneNotSupportedException e) {
91              System.err.println("Cannot clone MessageDigest");
92              try {
93                  digester = MessageDigest.getInstance(this.messageDigest.getAlgorithm());
94              }
95              catch (NoSuchAlgorithmException e1) {
96                  throw new RuntimeException(e1);
97              }
98          }
99  
100         final byte[] hashBytes = new byte[HASH_CODE_BYTE_SIZE];
101         this.deepDigest(hashBytes, digester, data);
102 
103 
104         final byte[] digest = digester.digest();
105         return Base64.encodeBase64URLSafeString(digest);
106     }

108     /**
109      * Tries to clone the {@link MessageDigest} that was created during construction. If the clone fails
110      * that is remembered and from that point on new {@link MessageDigest} instances will be created on
111      * every call.
112      * 
113      * @return Generates a {@link MessageDigest} to use during a call to {@link #generateKey(Object...)}
114      */
115     protected MessageDigest getMessageDigest() {
116         if (this.cloneNotSupported) {
117             final String algorithm = this.messageDigest.getAlgorithm();
118             try {
119                 return MessageDigest.getInstance(algorithm);
120             }
121             catch (NoSuchAlgorithmException e) {
122                 throw new IllegalStateException("MessageDigest algorithm '" + algorithm + "' was supported when " + this.getClass().getSimpleName() + " was created but is not now. This should not be possible.", e);
123             }
124         }
125         
126         try {
127             return (MessageDigest)this.messageDigest.clone();
128         }
129         catch (CloneNotSupportedException e) {
130             this.cloneNotSupported = true;
131             this.logger.warn("Could not clone MessageDigest using algorithm '" + this.messageDigest.getAlgorithm() + "'. MessageDigest.getInstance will be used from now on which will be much more expensive.", e);
132             return this.getMessageDigest();
133         }
134     }

136     /**
137      * Does a deep traversal, iterating over arrays, collections and maps and map entries. When
138      * a leaf in the object graph is found the appropriate digest method is called.
139      * 
140      * If {@link #setCheckforCycles(boolean)} is true and a cycle is found nothing is appeneded
141      * to the {@link MessageDigest} the second time the object is encountered.
142      * 
143      * @param hashBytes A byte array of length {@link #HASH_CODE_BYTE_SIZE} used when converting a hash code to byte array. Passed around to avoid extra object creation.
144      * @param messageDigest The {@link MessageDigest} used to generate the file hash key
145      * @param o The object to inspect
146      */
147     protected final void deepDigest(byte[] hashBytes, MessageDigest messageDigest, Object o) {
148         if (o == null) {
149             messageDigest.update(ZERO_AS_BYTES);
150             return;
151         }
152         
153         if (!register(o)) {
154             //Return without digesting anything in the case of a circular reference
155             return;
156         }
157         try {
158             if (o instanceof Class<?>) {
159                 this.digest(hashBytes, messageDigest, (Class<?>) o);
160             }
161             else if (o.getClass().isArray()) {
162                 final int length = Array.getLength(o);
163                 for (int index = 0; index < length; index++) {
164                     final Object arrayValue = Array.get(o, index);
165                     this.deepDigest(hashBytes, messageDigest, arrayValue);
166                 }
167             }
168             else if (o instanceof Iterable<?>) {
169                 for (final Object e : ((Iterable<?>) o)) {
170                     this.deepDigest(hashBytes, messageDigest, e);
171                 }
172             }
173             else if (o instanceof Map<?, ?>) {
174                 this.deepDigest(hashBytes, messageDigest, ((Map<?, ?>) o).entrySet());
175             }
176             else if (o instanceof Map.Entry<?, ?>) {
177                 final Map.Entry<?, ?> entry = (Map.Entry<?, ?>) o;
178                 this.deepDigest(hashBytes, messageDigest, entry.getKey());
179                 this.deepDigest(hashBytes, messageDigest, entry.getValue());
180             }
181             else {
182                 this.digest(hashBytes, messageDigest, o);
183             }
184         }
185         finally {
186             unregister(o);
187         }
188     }
189 
190     /**
191      * Special handling for digesting a {@link Class} which does not implement hashCode.
192      * 
193      * Default implmentation is to digest {@link Class#getName()}
194      */
195     protected void digest(byte[] hashBytes, MessageDigest messageDigest, Class<?> c) {
196         this.digest(hashBytes, messageDigest, c.getName());
197     }
198 
199     /**
200      * Add an object to the digest. Default adds the object's hashCode to the digest
201      */
202     protected void digest(byte[] hashBytes, MessageDigest messageDigest, Object o) {
203         intToBytes(hashBytes, o.hashCode());
204         messageDigest.update(hashBytes, 0, HASH_CODE_BYTE_SIZE);
205     }
206 
207     /**
208      * Convert an int to a byte array. The passed in byte[] must be at least
209      * 4 bytes long, if longer the bytes above the fourth are ignored.
210      * 
211      * @return the same byte[] that was passed in
212      */
213     protected final byte[] intToBytes(byte[] hashBytes, int value) {
214         hashBytes[0] = (byte) (value >>> 24);
215         hashBytes[1] = (byte) (value >>> 16);
216         hashBytes[2] = (byte) (value >>> 8);
217         hashBytes[3] = (byte) value;
218 
219         return hashBytes;
220     }
