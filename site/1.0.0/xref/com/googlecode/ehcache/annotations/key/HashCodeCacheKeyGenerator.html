19  import java.util.Arrays;
20  import java.util.Map;

22  /**
23   * This key generator is a good option for fast but rough comparison of method invocations. The standard Java
24   * hashCode method is used on the arguments to generate identity information. Essentially {@link Arrays#deepHashCode(Object[])}
25   * is used except with a long instead of an int for a hash accumulator to provide a larger key space. This is
26   * not a secure hashing algorithm.
27   * 
28   * <table>
29   *  <tr>
30   *      <th>Pros</th>
31   *      <th>Cons</th>
32   *  </tr>
33   *  <tr>
34   *      <td>
35   *          This is the fastest of the included key generation techniques by over 150%
36   *      </td>
37   *      <td>
38   *          The generated hash code is only a long so the key is only 64bits long. It is possible for different
39   *          arguments to have the same hash code which would result in a key collision.
40   *      </td>
41   *  </tr>
42   * </table>
43   * 
44   * @author Eric Dalquist
45   * @version $Revision$
46   */
47  public class HashCodeCacheKeyGenerator extends AbstractCacheKeyGenerator<Long> {
48      public static final String DEFAULT_BEAN_NAME = "com.googlecode.ehcache.annotations.key.HashCodeCacheKeyGenerator.DEFAULT_BEAN_NAME";
49      
50      protected static final int INITIAL_HASH = 1;
51      protected static final int MULTIPLIER = 31;
52      
53      public HashCodeCacheKeyGenerator() {
54      }

56      public HashCodeCacheKeyGenerator(boolean includeMethod, boolean includeParameterTypes) {
57          super(includeMethod, includeParameterTypes);
58      }

61      @Override
62      protected Long generateKey(Object... data) {
63          return this.deepHashCode(data);
64      }

66      /**
67       * @see Arrays#hashCode(long[])
68       */
69      protected long hashCode(long a[]) {
70          if (a == null)
71              return 0;
72  
73          long result = INITIAL_HASH;
74          for (final long element : a) {
75              int elementHash = (int)(element ^ (element >>> 32));
76              result = MULTIPLIER * result + elementHash;
77          }
78  
79          return result;
80      }

170         if (a == null)
171             return 0;
172 
173         long result = INITIAL_HASH;
174         for (final double element : a) {
175             long bits = Double.doubleToLongBits(element);
176             result = MULTIPLIER * result + (int)(bits ^ (bits >>> 32));
177         }
178         return result;
179     }

184     protected long deepHashCode(Object a[]) {
185         if (a == null)
186             return 0;
187 
188         long result = INITIAL_HASH;
189 
190         for (final Object element : a) {
191             final long elementHash = this.hashCode(element);
192             result = MULTIPLIER * result + elementHash;
193         }

216     protected long hashCode(Map.Entry<?, ?> e) {
217         if (e == null)
218             return 0;
219         
220         long result = INITIAL_HASH;
221 
222         final long keyHash = this.hashCode(e.getKey());
223         result = MULTIPLIER * result + keyHash;
224 
225         final long valueHash = this.hashCode(e.getValue());
226         result = MULTIPLIER * result + valueHash;
227 
228         return result;
229     }

231     protected final long hashCode(Object element) {
232         if (element == null || !register(element)) {
233             //Return 0 in place of the actual hash code in the case of a circular reference
234             return 0;
235         }
236         try {
237             long elementHash = 0;
238             if (element instanceof Object[])
239                 elementHash = deepHashCode((Object[]) element);
240             else if (element instanceof byte[])
241                 elementHash = hashCode((byte[]) element);
242             else if (element instanceof short[])
243                 elementHash = hashCode((short[]) element);
244             else if (element instanceof int[])
245                 elementHash = hashCode((int[]) element);
246             else if (element instanceof long[])
247                 elementHash = hashCode((long[]) element);
248             else if (element instanceof char[])
249                 elementHash = hashCode((char[]) element);
250             else if (element instanceof float[])
251                 elementHash = hashCode((float[]) element);
252             else if (element instanceof double[])
253                 elementHash = hashCode((double[]) element);
254             else if (element instanceof boolean[])
255                 elementHash = hashCode((boolean[]) element);
256             else if (element instanceof Class<?>)
257                 elementHash = getHashCode((Class<?>)element);
258             else if (element instanceof Iterable<?>)
259                 elementHash = deepHashCode((Iterable<?>)element);
260             else if (element instanceof Map<?, ?>)
261                 elementHash = deepHashCode(((Map<?, ?>)element).entrySet());
262             else if (element instanceof Map.Entry<?, ?>)
263                 elementHash = hashCode((Map.Entry<?, ?>)element);
264             else
265                 elementHash = getHashCode(element);
266             return elementHash;
267         }
268         finally {
269             unregister(element);
270         }
271     }

273     /**
274      * Generate hash code for an Class which by default uses object identity hash codes
275      */
276     protected long getHashCode(Class<?> c) {
277         return c.getName().hashCode();
278     }

281      * Generate hash code for an object
282      */

284         return o.hashCode();
285     }
