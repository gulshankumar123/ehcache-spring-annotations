149     private MethodAttribute getMethodAttribute(final Method method, final Class<?> targetClass) {
150         //Check if the method has already been inspected and should be ignored
151         if (this.ingoredMethods.containsKey(targetClass, method)) {
152             return null;
153         }
154         
155         //Check the cache if the method has already had its advise attributes created
156         final MethodAttribute attributes = this.attributesCache.get(targetClass, method);
157         if (attributes != null) {
158             return attributes;
159         }
160         
161         // We need to work it out.
162         final MethodAttribute att = this.computeMethodAttribute(method, targetClass);
163         
164         // Put it in the cache.
165         if (att == null) {
166             this.ingoredMethods.put(targetClass, method, Boolean.TRUE);
167         }
168         else  {
169             this.logger.debug("Adding {} advised method '{}' with attribute: {}", new Object[] { att.getAdviceType(), method.getName(), att });
170             final MethodAttribute existing = this.attributesCache.putIfAbsent(targetClass, method, att);
171             if (existing != null) {
172                 return existing;
173             }
174         }
175         
176         return att;
177     }

195     /**
196      * Get or create the specified cache if it does not exist and createCaches is set to true. 
197      * 
198      * @param cacheName The name of the cache to retrieve
199      * @return The cache
200      * @throws CacheNotFoundException if the cache does not exist and createCaches is false.
201      */
202     protected Ehcache getCache(String cacheName) {
203         final CacheManager cacheManager = this.getCacheManager();
204         
205         Ehcache cache = cacheManager.getEhcache(cacheName);
206         if (cache == null) {
207             if (this.createCaches) {
208                 this.logger.warn("No cache named '{}' exists, it will be created from the defaultCache", cacheName);
209                 try {
210                     cacheManager.addCache(cacheName);
211                 }
212                 catch (ObjectExistsException oee) {
213                     this.logger.trace("Race condition creating missing cache '{}', ignoring and retrieving existing cache", cacheName);
214                 }
215                 cache = cacheManager.getEhcache(cacheName);
216             }
217             else {
218                 throw new CacheNotFoundException(cacheName);
219             }
220         }
221         return cache;
222     }

224     private MethodAttribute computeMethodAttribute(Method method, Class<?> targetClass) {
225         // Don't allow no-public methods as required.
226         if (this.allowPublicMethodsOnly() && !Modifier.isPublic(method.getModifiers())) {
227             return null;
228         }
229 
230         // The method may be on an interface, but we need attributes from the target class.
231         // If the target class is null, the method will be unchanged.
232         Method specificMethod = ClassUtils.getMostSpecificMethod(method, targetClass);
233         // If we are dealing with method with generic parameters, find the original method.
234         specificMethod = BridgeMethodResolver.findBridgedMethod(specificMethod);
235 
236         // First try is the method in the target class.
237         MethodAttribute att = this.findMethodAttribute(specificMethod);
238         if (att != null) {
239             return att;
240         }
241 
242         if (specificMethod != method) {
243             // Fallback is to look at the original method.
244             att = this.findMethodAttribute(method);
245             if (att != null) {
246                 return att;
247             }
248         }
249         
250         return null;
251     }

253     /**
254      * Determine if the specified {@link AnnotatedElement} is annotated with either {@link Cacheable} or {@link TriggersRemove}
255      * 
256      * @param method The element to inspect
257      * @return The advice attributes about the element, null if the element is not advised
258      */
259     private MethodAttribute findMethodAttribute(Method method) {
260         Cacheable cacheableAnnotation = method.getAnnotation(Cacheable.class);
261         if (cacheableAnnotation != null) {
262             final ParameterMask parameterMask = this.parsePartialCacheKeyAnnotations(method);
263             return this.parseCacheableAnnotation(cacheableAnnotation, parameterMask);
264         }
265         
266         TriggersRemove triggersRemove = method.getAnnotation(TriggersRemove.class);
267         if (triggersRemove != null) {
268             final ParameterMask parameterMask = this.parsePartialCacheKeyAnnotations(method);
269             return this.parseTriggersRemoveAnnotation(triggersRemove, parameterMask);
270         }
271         
272         for (final Annotation metaAnn : method.getAnnotations()) {
273             final Class<? extends Annotation> annotationType = metaAnn.annotationType();
274             cacheableAnnotation = annotationType.getAnnotation(Cacheable.class);
275             if (cacheableAnnotation != null) {
276                 final ParameterMask parameterMask = this.parsePartialCacheKeyAnnotations(method);
277                 return this.parseCacheableAnnotation(cacheableAnnotation, parameterMask);
278             }
279             
280             triggersRemove = annotationType.getAnnotation(TriggersRemove.class);
281             if (triggersRemove != null) {
282                 final ParameterMask parameterMask = this.parsePartialCacheKeyAnnotations(method);
283                 return this.parseTriggersRemoveAnnotation(triggersRemove, parameterMask);
284             }
285         }
286 
287         return null;
288     }

290     /**
291      * Parse the parameters annotated with {@link PartialCacheKey}.
292      * 
293      * @param method The method who's parameters should be checked.
294      * @return The parameter mask for the annotated parameters.
295      */
296     protected ParameterMask parsePartialCacheKeyAnnotations(Method method) {
297         final Annotation[][] parameterAnnotations = method.getParameterAnnotations();
298         final boolean[] mask = new boolean[parameterAnnotations.length];
299         for (int i = 0; i < parameterAnnotations.length; i++) {
300             final Annotation[] annotations = parameterAnnotations[i];
301             for (final Annotation annotation : annotations) {
302                 final Class<? extends Annotation> annotationType = annotation.annotationType();
303                 if (PartialCacheKey.class.equals(annotationType)) {
304                     mask[i] = true;
305                     break;
306                 }
307             }
308         }
309         
310         return ParameterMask.create(mask);
311     }

313     /**
314      * Construct a {@link CacheableAttribute} from a {@link Cacheable} annotation.
315      * 
316      * @param ann The annotation to build the attributes from
317      * @return The constructed cacheable advise attributes
318      */
319     protected CacheableAttribute parseCacheableAnnotation(Cacheable ann, ParameterMask parameterMask) {
320         Ehcache cache = this.getCache(ann.cacheName());
321         ThreadLocal<MethodInvocation> entryFactory = null;
322         if (ann.selfPopulating()) {
323             final SelfPopulatingCacheTracker selfPopulatingCacheTracker = this.createSelfPopulatingCacheInternal(cache);
324             cache = selfPopulatingCacheTracker.selfPopulatingCache;
325             entryFactory = selfPopulatingCacheTracker.cacheEntryFactory;
326         }
327         
328         final Ehcache exceptionCache;
329         if (StringUtils.hasLength(ann.exceptionCacheName())) {
330             exceptionCache = this.getCache(ann.exceptionCacheName());
331         }
332         else {
333             exceptionCache = null;
334         }
335         
336         final String keyGeneratorName = ann.keyGeneratorName();
337         final KeyGenerator keyGenerator = ann.keyGenerator();
338         final CacheKeyGenerator<? extends Serializable> cacheKeyGenerator = this.getCacheKeyGenerator(keyGeneratorName, keyGenerator);
339         
340         return new CacheableAttributeImpl(cache, exceptionCache, cacheKeyGenerator, parameterMask, entryFactory);
341     }

343     /**
344      * Creates or retrieves a SelfPopulatingCacheTracker for the specified cache depending on the
345      * configured {@link SelfPopulatingCacheScope}
346      * 
347      * @param cache The cache to create a self populating instance of
348      * @return The SelfPopulatingCache and corresponding factory object to use
349      */
350     protected final SelfPopulatingCacheTracker createSelfPopulatingCacheInternal(Ehcache cache) {
351         //If method scoped just create a new instance 
352         if (SelfPopulatingCacheScope.METHOD == this.selfPopulatingCacheScope) {
353             return this.createSelfPopulatingCache(cache);
354         }
355 
356         //Shared scope, try loading the instance from local Map
357         
358         //See if there is a cached SelfPopulatingCache for the name
359         final String cacheName = cache.getName();
360         SelfPopulatingCacheTracker selfPopulatingCacheTracker = this.selfPopulatingCaches.get(cacheName);
361         if (selfPopulatingCacheTracker == null) {
362             selfPopulatingCacheTracker = this.createSelfPopulatingCache(cache);
363             
364             //do putIfAbsent to handle concurrent creation. If a value is returned it was already put and that
365             //value should be used. If no value was returned the newly created selfPopulatingCache should be used
366             final SelfPopulatingCacheTracker existing = this.selfPopulatingCaches.putIfAbsent(cacheName, selfPopulatingCacheTracker);
367             if (existing != null) {
368                 selfPopulatingCacheTracker = existing;
369             }
370         }
371         
372         return selfPopulatingCacheTracker;
373     }

385      * Construct a {@link TriggersRemoveAttribute} from a {@link TriggersRemove} annotation.
386      * 
387      * @param ann The annotation to build the attributes from
388      * @return The constructed triggers remove advise attributes
389      */
390     protected TriggersRemoveAttribute parseTriggersRemoveAnnotation(TriggersRemove ann, ParameterMask parameterMask) {
391         final String[] cacheNames = ann.cacheName();
392         final List<Ehcache> caches = new ArrayList<Ehcache>(cacheNames.length);
393         for (final String cacheName : cacheNames) {
394             final Ehcache cache = this.getCache(cacheName);
395             caches.add(cache);
396         }
397 
398         final String keyGeneratorName = ann.keyGeneratorName();
399         final KeyGenerator keyGenerator = ann.keyGenerator();
400         final CacheKeyGenerator<? extends Serializable> cacheKeyGenerator = this.getCacheKeyGenerator(keyGeneratorName, keyGenerator);
401         
402         return new TriggersRemoveAttributeImpl(caches, cacheKeyGenerator, parameterMask, ann.removeAll(), ann.when());
403     }

406      * Get the {@link CacheKeyGenerator} by name. Returning a default generator if the name is empty or null
407      * 
408      * @param keyGeneratorName Name of the generator to retrieve
409      * @return The named generator or the default generator if the name was empty or null
410      */
411     @SuppressWarnings("unchecked")
412     protected final CacheKeyGenerator<? extends Serializable> getCacheKeyGenerator(String keyGeneratorName, KeyGenerator keyGenerator) {
413         String keyGeneratorClassName = keyGenerator.name();
414         if (keyGeneratorClassName.length() > 0) {
415             return this.getOrCreateCacheKeyGenerator(keyGenerator);
416         }
417 
418         final CacheKeyGenerator<? extends Serializable> cacheKeyGenerator;
419         if (StringUtils.hasLength(keyGeneratorName)) {
420             cacheKeyGenerator = this.beanFactory.getBean(keyGeneratorName, CacheKeyGenerator.class);
421         } else {
422             cacheKeyGenerator = this.defaultCacheKeyGenerator;
423         }
424         return cacheKeyGenerator;
425     }

427     @SuppressWarnings("unchecked")
428     protected final CacheKeyGenerator<? extends Serializable> getOrCreateCacheKeyGenerator(KeyGenerator keyGenerator) {
429         final StringBuilder beanNameBuilder = new StringBuilder();
430         
431         String keyGeneratorClassName = keyGenerator.name();
432         if (!keyGeneratorClassName.contains(".")) {
433             keyGeneratorClassName  = "com.googlecode.ehcache.annotations.key." + keyGeneratorClassName;
434         }
435         
436         beanNameBuilder.append(keyGeneratorClassName);
437         
438         final MutablePropertyValues mutablePropertyValues = new MutablePropertyValues();
439         
440         //Sort the properties array first so bean name generation is always consistent
441         final Property[] properties = keyGenerator.properties();
442         Arrays.sort(properties, PropertyComparator.INSTANCE);
443         
444         for (Property property : properties) {
445             final String name = property.name();
446             final String value = property.value();
447             final String ref = property.ref();
448             
449             beanNameBuilder.append("[").append(name).append(",").append(value).append(",").append(ref).append("]");
450             
451             if (value.length() > 0) {
452                 if (ref.length() > 0) {
453                     throw new IllegalArgumentException("Only one of value or ref must be specified no both on Property with name: " + name);
454                 }
455                 
456                 mutablePropertyValues.addPropertyValue(name, value);
457             }
458             else if (ref.length() > 0) {
459                 mutablePropertyValues.addPropertyValue(name, new RuntimeBeanReference(ref));
460             }
461             else {
462                 throw new IllegalArgumentException("Either value or ref must be specified on Property with name: " + name);
463             }
464         }
465         
466         final String beanName = beanNameBuilder.toString();
467         
468         //See if the generator is already registered using the compiled bean name, if so just use that instance
469         if (this.cacheKeyBeanFactory.containsBean(beanName)) {
470             return this.cacheKeyBeanFactory.getBean(beanName, CacheKeyGenerator.class);
471         }
472 
473         //Create and register the bean if it didn't already exist
474         final AbstractBeanDefinition beanDefinition;
475         try {
476             beanDefinition = BeanDefinitionReaderUtils.createBeanDefinition(null, keyGeneratorClassName, ClassUtils.getDefaultClassLoader());
477         }
478         catch (ClassNotFoundException e) {
479             throw new BeanCreationException("Could not find class '" + keyGeneratorClassName + "' to create CacheKeyGenerator from", e);
480         }
481         
482         if (ReflectionHelperAware.class.isAssignableFrom(beanDefinition.getBeanClass())) {
483             final RuntimeBeanReference cacheManagerReference = new RuntimeBeanReference(AnnotationDrivenEhCacheBeanDefinitionParser.CACHING_REFLECTION_HELPER_BEAN_NAME);
484             mutablePropertyValues.addPropertyValue("reflectionHelper", cacheManagerReference);
485         }
486         
487         beanDefinition.setPropertyValues(mutablePropertyValues);
488         this.cacheKeyBeanFactory.registerBeanDefinition(beanName, beanDefinition);
489         
490         return this.cacheKeyBeanFactory.getBean(beanName, CacheKeyGenerator.class);
491     }

503     /**
504      * EhCache entry factory that uses a ThreadLocal to pass a MethodInvocation into the factory
505      * for object creation.
506      */
507     private static class ThreadLocalCacheEntryFactory implements CacheEntryFactory {
508         public final ThreadLocal<MethodInvocation> entryFactory = new ThreadLocal<MethodInvocation>();
509 
510         public Object createEntry(Object arg0) throws Exception {
511             final MethodInvocation methodInvocation = this.entryFactory.get();
512             if (methodInvocation == null) {
513                 throw new RuntimeException("No MethodInvocation specified in the ThreadLocal");
514             }
515             
516             try {
517                 return methodInvocation.proceed();
518             }
519             catch (Throwable t) {
520                 if (t instanceof Exception) {
521                     throw (Exception)t;
522                 }
523                 else if (t instanceof Error) {
524                     throw (Error)t;
525                 }
526                 
527                 throw new Exception(t);
528             }
529         }
