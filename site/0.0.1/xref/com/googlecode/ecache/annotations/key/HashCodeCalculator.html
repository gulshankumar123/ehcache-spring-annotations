17  package com.googlecode.ecache.annotations.key;
18  
19  /**
20   * <p>
21   * Builds the checksum and hash code needed to create a
22   * <code>{@link HashCodeCacheKey}</code>.
23   * </p>
24   *
25   * @author Omar Irbouh
26   * @author Alex Ruiz
27   */
28  public final class HashCodeCalculator {
29  
30    private static final int INITIAL_HASH = 17;
31  
32    private static final int MULTIPLIER = 37;
33  
34    private long checkSum;
35  
36    /**
37     * Counts the number of times <code>{@link #append(int)}</code> is executed.
38     * It is also used to build <code>{@link #checkSum}</code> and
39     * <code>{@link #hashCode}</code>.
40     */
41    private int count;
42  
43    /**
44     * Hash code to build;
45     */
46    private int hashCode;
47  
48    /**
49     * Constructor.
50     */
51    public HashCodeCalculator() {
52      super();
53      hashCode = INITIAL_HASH;
54    }
55  
56    /**
57     * Recalculates <code>{@link #checkSum}</code> and
58     * <code>{@link #hashCode}</code> using the specified value.
59     *
60     * @param value
61     *          the specified value.
62     */
63    public void append(int value) {
64      count++;
65      int valueToAppend = count * value;
66  
67      hashCode = MULTIPLIER * hashCode + (valueToAppend ^ (valueToAppend >>> 16));
68      checkSum += valueToAppend;
69    }
70  
71    /**
72     * @return the number that ensures that the combination hashCode/checSum is
73     *         unique
74     */
75    public long getCheckSum() {
76      return checkSum;
77    }
78  
79    /**
80     * @return the calculated hash code
81     */
82    public int getHashCode() {
83      return hashCode;
84    }
85  }
