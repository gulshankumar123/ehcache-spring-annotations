36  /**
37   * {@link BeanDefinitionParser} that sets up {@link DefaultBeanFactoryPointcutAdvisor}
38   * instances to wrap {@link EhCacheInterceptor}s around {@link Cacheable}s and {@link TriggersRemove}
39   * advised methods.
40   * 
41   * @author Nicholas Blair, nblair@doit.wisc.edu
42   * @version $Id$
43   */
44  public class AnnotationDrivenEhCacheBeanDefinitionParser implements BeanDefinitionParser {
45  
46      public static final String EHCACHE_CACHING_ADVISOR_BEAN_NAME = "com.googlecode.ecache.annotations.config.internalEhCacheCachingAdvisor";
47      
48      public static final String DEFAULT_CACHE_KEY_GENERATOR = HashCodeCacheKeyGenerator.class.getName() + "_DEFFAULT";
49      
50      /* (non-Javadoc)
51       * @see org.springframework.beans.factory.xml.BeanDefinitionParser#parse(org.w3c.dom.Element, org.springframework.beans.factory.xml.ParserContext)
52       */
53      public BeanDefinition parse(Element element, ParserContext parserContext) {
54          AopNamespaceUtils.registerAutoProxyCreatorIfNecessary(parserContext, element);
55          if (!parserContext.getRegistry().containsBeanDefinition(EHCACHE_CACHING_ADVISOR_BEAN_NAME)) {
56              Object elementSource = parserContext.extractSource(element);
57              
58              RootBeanDefinition cacheAttributeSource = new RootBeanDefinition(CacheAttributeSourceImpl.class);
59              cacheAttributeSource.setSource(elementSource);
60              cacheAttributeSource.setRole(BeanDefinition.ROLE_INFRASTRUCTURE);
61              cacheAttributeSource.getPropertyValues().add("cacheManagerBeanName", element.getAttribute("cache-manager"));
62              cacheAttributeSource.getPropertyValues().add("createCaches", Boolean.parseBoolean(element.getAttribute("create-missing-caches")));
63              String cacheableAttributeSourceBeanName = parserContext.getReaderContext().registerWithGeneratedName(cacheAttributeSource);
64              RuntimeBeanReference cacheableAttributeSourceRuntimeReference = new RuntimeBeanReference(cacheableAttributeSourceBeanName);
65              
66              RootBeanDefinition cacheablePointcutSource = new RootBeanDefinition(CacheStaticMethodMatcherPointcut.class);
67              cacheablePointcutSource.setSource(elementSource);
68              cacheablePointcutSource.setRole(BeanDefinition.ROLE_INFRASTRUCTURE);
69              cacheablePointcutSource.getPropertyValues().add("cacheAttributeSource", cacheableAttributeSourceRuntimeReference);
70              String cacheablePointcutBeanName = parserContext.getReaderContext().registerWithGeneratedName(cacheablePointcutSource);
71              
72              RootBeanDefinition cachingInterceptorSource = new RootBeanDefinition(EhCacheInterceptor.class);
73              cachingInterceptorSource.setSource(elementSource);
74              cachingInterceptorSource.setRole(BeanDefinition.ROLE_INFRASTRUCTURE);
75              cachingInterceptorSource.getPropertyValues().add("cacheAttributeSource", cacheableAttributeSourceRuntimeReference);
76              String cachingInterceptorBeanName = parserContext.getReaderContext().registerWithGeneratedName(cachingInterceptorSource);
77              
78              
79              RootBeanDefinition cachingPointcutAdvisorSource = new RootBeanDefinition(DefaultBeanFactoryPointcutAdvisor.class);
80              cachingPointcutAdvisorSource.setSource(elementSource);
81              cachingPointcutAdvisorSource.setRole(BeanDefinition.ROLE_INFRASTRUCTURE);
82              cachingPointcutAdvisorSource.getPropertyValues().add("adviceBeanName", cachingInterceptorBeanName);
83              cachingPointcutAdvisorSource.getPropertyValues().add("pointcut", new RuntimeBeanReference(cacheablePointcutBeanName));
84              parserContext.getRegistry().registerBeanDefinition(EHCACHE_CACHING_ADVISOR_BEAN_NAME, cachingPointcutAdvisorSource);
85              
86              
87              final RootBeanDefinition defaultKeyGenerator = new RootBeanDefinition(HashCodeCacheKeyGenerator.class);
88              defaultKeyGenerator.setSource(elementSource);
89              defaultKeyGenerator.setRole(BeanDefinition.ROLE_INFRASTRUCTURE);
90              parserContext.getRegistry().registerBeanDefinition(DEFAULT_CACHE_KEY_GENERATOR, defaultKeyGenerator);
91             
92          }
93          return null;
94      }
