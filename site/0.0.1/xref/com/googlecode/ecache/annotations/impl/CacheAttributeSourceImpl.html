125     private MethodAttribute getMethodAttribute(final Method method, final Class<?> targetClass) {
126         final Object cacheKey = this.getCacheKey(method, targetClass);
127         
128         //Check if the method has already been inspected and should be ignored
129         if (this.ingoredMethods.containsKey(cacheKey)) {
130             return null;
131         }
132         
133         //Check the cache if the method has already had its advise attributes created
134         final MethodAttribute attributes = this.attributesCache.get(cacheKey);
135         if (attributes != null) {
136             return attributes;
137         }
138         
139         // We need to work it out.
140         final MethodAttribute att = this.computeMethodAttribute(method, targetClass);
141         
142         // Put it in the cache.
143         if (att == null) {
144             this.ingoredMethods.put(cacheKey, cacheKey);
145         }
146         else  {
147             this.logger.debug("Adding {} advised method '{}' under key '{}' with attribute: {}", new Object[] { att.getAdviceType(), method.getName(), cacheKey, att });
148             final MethodAttribute existing = this.attributesCache.putIfAbsent(cacheKey, att);
149             if (existing != null) {
150                 return existing;
151             }
152         }
153         
154         return att;
155     }

197     /**
198      * Get or create the specified cache if it does not exist and createCaches is set to true. 
199      * 
200      * @param cacheName The name of the cache to retrieve
201      * @return The cache
202      * @throws CacheNotFoundException if the cache does not exist and createCaches is false.
203      */
204     protected Ehcache getCache(String cacheName) {
205         final CacheManager cacheManager = this.getCacheManager();
206         
207         Ehcache cache = cacheManager.getCache(cacheName);
208         if (cache == null) {
209             if (this.createCaches) {
210                 this.logger.warn("No cache named '{}' exists, it will be created from the defaultCache", cacheName);
211                 try {
212                     cacheManager.addCache(cacheName);
213                 }
214                 catch (ObjectExistsException oee) {
215                     this.logger.trace("Race condition creating missing cache '{}', ignoring and retrieving existing cache", cacheName);
216                 }
217                 cache = cacheManager.getCache(cacheName);
218             }
219             else {
220                 throw new CacheNotFoundException(cacheName);
221             }
222         }
223         return cache;
224     }

226     private MethodAttribute computeMethodAttribute(Method method, Class<?> targetClass) {
227         // Don't allow no-public methods as required.
228         if (this.allowPublicMethodsOnly() && !Modifier.isPublic(method.getModifiers())) {
229             return null;
230         }
231 
232         // The method may be on an interface, but we need attributes from the target class.
233         // If the target class is null, the method will be unchanged.
234         Method specificMethod = ClassUtils.getMostSpecificMethod(method, targetClass);
235         // If we are dealing with method with generic parameters, find the original method.
236         specificMethod = BridgeMethodResolver.findBridgedMethod(specificMethod);
237 
238         // First try is the method in the target class.
239         MethodAttribute att = this.findMethodAttribute(specificMethod);
240         if (att != null) {
241             return att;
242         }
243 
244         if (specificMethod != method) {
245             // Fallback is to look at the original method.
246             att = this.findMethodAttribute(method);
247             if (att != null) {
248                 return att;
249             }
250         }
251         
252         return null;
253     }

255     /**
256      * Determine if the specified {@link AnnotatedElement} is annotated with either {@link Cacheable} or {@link TriggersRemove}
257      * 
258      * @param ae The element to inspect
259      * @return The advice attributes about the element, null if the element is not advised
260      */
261     protected MethodAttribute findMethodAttribute(AnnotatedElement ae) {
262         Cacheable cacheableAnnotation = ae.getAnnotation(Cacheable.class);
263         if (cacheableAnnotation != null) {
264             return this.parseCacheableAnnotation(cacheableAnnotation);
265         }
266         
267         TriggersRemove triggersRemove = ae.getAnnotation(TriggersRemove.class);
268         if (triggersRemove != null) {
269             return this.parseTriggersRemoveAnnotation(triggersRemove);
270         }
271         
272         for (final Annotation metaAnn : ae.getAnnotations()) {
273             final Class<? extends Annotation> annotationType = metaAnn.annotationType();
274             cacheableAnnotation = annotationType.getAnnotation(Cacheable.class);
275             if (cacheableAnnotation != null) {
276                 return this.parseCacheableAnnotation(cacheableAnnotation);
277             }
278             
279             triggersRemove = annotationType.getAnnotation(TriggersRemove.class);
280             if (triggersRemove != null) {
281                 return this.parseTriggersRemoveAnnotation(triggersRemove);
282             }
283         }
284 
285         return null;
286     }

288     /**
289      * Construct a {@link CacheableAttribute} from a {@link Cacheable} annotation.
290      * 
291      * @param ann The annotation to build the attributes from
292      * @return The constructed cacheable advise attributes
293      */
294     protected CacheableAttribute parseCacheableAnnotation(Cacheable ann) {
295         Ehcache cache = this.getCache(ann.cacheName());
296         ThreadLocal<MethodInvocation> entryFactory = null;
297         if (ann.selfPopulating()) {
298             final ThreadLocalCacheEntryFactory cacheEntryFactory = new ThreadLocalCacheEntryFactory();
299             entryFactory = cacheEntryFactory.entryFactory;
300             cache = new SelfPopulatingCache(cache, cacheEntryFactory);
301         }
302         
303         final Ehcache exceptionCache;
304         if (StringUtils.hasLength(ann.exceptionCacheName())) {
305             exceptionCache = this.getCache(ann.exceptionCacheName());
306         }
307         else {
308             exceptionCache = null;
309         }
310         
311         final String keyGeneratorName = ann.keyGeneratorName();
312         final CacheKeyGenerator cacheKeyGenerator = getCacheKeyGenerator(keyGeneratorName);
313         
314         return new CacheableAttributeImpl(cache, exceptionCache, cacheKeyGenerator, entryFactory);
315     }

320      * @param ann The annotation to build the attributes from

324         final Ehcache cache = this.getCache(ann.cacheName());
325 
326         final String keyGeneratorName = ann.keyGeneratorName();
327         final CacheKeyGenerator cacheKeyGenerator = getCacheKeyGenerator(keyGeneratorName);

329         return new TriggersRemoveAttributeImpl(cache, cacheKeyGenerator, ann.removeAll());

349     /**
350      * EhCache entry factory that uses a ThreadLocal to pass a MethodInvocation into the factory
351      * for object creation.
352      */
353     private static class ThreadLocalCacheEntryFactory implements CacheEntryFactory {
354         public final ThreadLocal<MethodInvocation> entryFactory = new ThreadLocal<MethodInvocation>();
355 
356         public Object createEntry(Object arg0) throws Exception {
357             final MethodInvocation methodInvocation = this.entryFactory.get();
358             if (methodInvocation == null) {
359                 throw new RuntimeException("No MethodInvocation specified in the ThreadLocal");
360             }
361             
362             try {
363                 return methodInvocation.proceed();
364             }
365             catch (Throwable t) {
366                 if (t instanceof Exception) {
367                     throw (Exception)t;
368                 }
369                 else if (t instanceof Error) {
370                     throw (Error)t;
371                 }
372                 
373                 throw new Exception(t);
374             }
375         }
