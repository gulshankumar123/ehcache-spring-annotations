38  public class ConfigurationFailureTest {

40  	/**
41  	 * Test verifies behavior when no {@link CacheManager} is defined in the Spring configuration.
42  	 */
43      @Test
44      public void testNoCacheManager() {
45          try {
46              new ClassPathXmlApplicationContext("/noCacheManagerTestContext.xml");
47              Assert.fail("Test should have failed with no CacheManager defined");
48          }
49          catch (BeanCreationException bce) {
50              Assert.assertEquals("cacheableTestImpl", bce.getBeanName());
51              final NoSuchBeanDefinitionException nsbd = (NoSuchBeanDefinitionException)bce.getCause();
52              Assert.assertEquals("cacheManager", nsbd.getBeanName());
53          }
54      }
55      
56      /**
57       * Test verifies behavior when no {@link Ehcache} is defined for
58       * the cacheName attribute on the  {@link Cacheable} annotated method AND
59       * 'createMissingCaches' is false.
60       */
61      @Test
62      public void testNoCache() {
63          try {
64              new ClassPathXmlApplicationContext("/noCacheTestContext.xml");
65              Assert.fail("Test should have failed with no Cache defined");
66          }
67          catch (BeanCreationException bce) {
68              Assert.assertEquals("missingCacheNameImpl", bce.getBeanName());
69              final CacheNotFoundException cnfe = (CacheNotFoundException)bce.getCause();
70              Assert.assertEquals("nonexistent", cnfe.getCacheName());
71          }
72      }
73      
74      /**
75       * Test verifies behavior when no {@link CacheKeyGenerator} is defined for
76       * the cacheKeyGenerator attribute on the {@link Cacheable} annotated method.
77       */
78      @Test
79      public void testNoCacheKeyGenerator() {
80          try {
81              new ClassPathXmlApplicationContext("/noCacheKeyGeneratorTestContext.xml");
82              Assert.fail("Test should have failed with no CacheKeyGenerator defined");

86              final NoSuchBeanDefinitionException nsbd = (NoSuchBeanDefinitionException)bce.getCause();
