47      /**
48       * Validate an empty evict-expired-elements Element.
49       */
50      @Test
51      public void testParseEmptyEvictExpiredElement() {
52          NodeList childNodes = EasyMock.createMock(NodeList.class);
53          EasyMock.expect(childNodes.getLength()).andReturn(0);
54          
55          Element evictExpired = EasyMock.createMock(Element.class);
56          EasyMock.expect(evictExpired.getChildNodes()).andReturn(childNodes);
57          EasyMock.replay(childNodes, evictExpired);
58          
59          EhCacheConfigBeanDefinitionParser parser = new EhCacheConfigBeanDefinitionParser();
60          List<CacheNameMatcher> matchers = parser.parseEvictExpiredElement(evictExpired);
61          Assert.assertNotNull(matchers);
62          Assert.assertEquals(1, matchers.size());
63          CacheNameMatcher matcher = matchers.get(0);
64          Assert.assertTrue(matcher instanceof PatternCacheNameMatcherImpl);
65          PatternCacheNameMatcherImpl casted = (PatternCacheNameMatcherImpl) matcher;
66          Assert.assertEquals(".*", casted.getPattern().toString());
67          
68          EasyMock.verify(childNodes, evictExpired);
69      }

106     /**
107      * Validate an evict-expired-elements Element with a single exclude element.
108      * Presence of exclude element will prepend a Include All matcher to the front of the list.
109      */
110     @Test
111     public void testParseEvictExpiredSingleExclude() {
112         Node nameAttribute = EasyMock.createMock(Node.class);
113         EasyMock.expect(nameAttribute.getTextContent()).andReturn("foo");
114         NamedNodeMap excludeAttributes = EasyMock.createMock(NamedNodeMap.class);
115         EasyMock.expect(excludeAttributes.getNamedItem(EhCacheConfigBeanDefinitionParser.XSD_ATTRIBUTE__NAME)).andReturn(nameAttribute);
116         
117         Node excludeNode = EasyMock.createMock(Node.class);
118         EasyMock.expect(excludeNode.getNodeType()).andReturn(Node.ELEMENT_NODE);
119         EasyMock.expect(excludeNode.getLocalName()).andReturn("exclude");
120         EasyMock.expect(excludeNode.getAttributes()).andReturn(excludeAttributes);
121         
122         NodeList childNodes = EasyMock.createMock(NodeList.class);
123         EasyMock.expect(childNodes.getLength()).andReturn(1);
124         EasyMock.expect(childNodes.item(0)).andReturn(excludeNode);
125         
126         Element evictExpired = EasyMock.createMock(Element.class);
127         EasyMock.expect(evictExpired.getChildNodes()).andReturn(childNodes);
128         EasyMock.replay(nameAttribute, excludeAttributes, excludeNode, childNodes, evictExpired);
129         
130         EhCacheConfigBeanDefinitionParser parser = new EhCacheConfigBeanDefinitionParser();
131         List<CacheNameMatcher> matchers = parser.parseEvictExpiredElement(evictExpired);
132         Assert.assertNotNull(matchers);
133         Assert.assertEquals(2, matchers.size());
134         Assert.assertEquals(EhCacheConfigBeanDefinitionParser.INCLUDE_ALL_CACHE_NAME_MATCHER, matchers.get(0));
135         
136         CacheNameMatcher matcher = matchers.get(1);
137         Assert.assertTrue(matcher instanceof NotCacheNameMatcherImpl);
138         NotCacheNameMatcherImpl casted = (NotCacheNameMatcherImpl) matcher;
139         ExactCacheNameMatcherImpl wrapped = (ExactCacheNameMatcherImpl) casted.getWrapped();
140         Assert.assertEquals("foo", wrapped.getName());
141         
142         EasyMock.verify(nameAttribute, excludeAttributes, excludeNode, childNodes, evictExpired);
143     }
