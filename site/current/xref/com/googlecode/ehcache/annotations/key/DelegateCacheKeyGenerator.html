17  package com.googlecode.ehcache.annotations.key;
18  
19  import java.io.Serializable;
20  
21  
22  /**
23   * {@link AbstractCacheKeyGenerator} implementation designed for subclassing.
24   * 
25   * Subclasses can implement the {@link #getAdditionalDataForKeyGeneration()} method to
26   * seed additional data into the call to {@link #generateKey(Object...)}.
27   * 
28   * @author Nicholas Blair
29   * @version $Id$
30   */
31  public abstract class DelegateCacheKeyGenerator extends AbstractCacheKeyGenerator<Serializable> {
32  
33      private final AbstractCacheKeyGenerator<? extends Serializable> delegate;
34      
35      /**
36       * @param delegate
37       */
38      public DelegateCacheKeyGenerator(
39              AbstractCacheKeyGenerator<? extends Serializable> delegate) {
40          super();
41          this.delegate = delegate;
42      }
43  
44      /**
45       * Calls {@link #generateKey(Object...)} on the delegate passing in data argument AND the result
46       * from {@link #getAdditionalDataForKeyGeneration()}.
47       * 
48       * @see #getAdditionalDataForKeyGeneration()
49       * @see com.googlecode.ehcache.annotations.key.AbstractCacheKeyGenerator#generateKey(java.lang.Object[])
50       */
51      @Override
52      public final Serializable generateKey(Object... data) {
53          this.setCheckforCycles(delegate.isCheckforCycles());
54          return this.delegate.generateKey(getAdditionalDataForKeyGeneration(), data);
55      }
56  
57      /**
58       * 
59       * @return a {@link Serializable} object to inject into the calls to {@link #generateKey(Object...)}
60       */
61      public abstract Serializable getAdditionalDataForKeyGeneration();
62  }
