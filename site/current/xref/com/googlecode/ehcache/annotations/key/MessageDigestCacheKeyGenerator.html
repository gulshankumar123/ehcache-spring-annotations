30  /**
31   * This key generator is a good option when a secure hash of the arguments is needed or when a
32   * larger final key space is desired. As with {@link HashCodeCacheKeyGenerator} the argument
33   * hashCode is used for key generation data but instead of the simple Java hash algorithm being
34   * used any {@link MessageDigest} can be used, SHA-1 is used by default. The generated hash is
35   * base64 encoded using {@link Base64#encodeBase64URLSafeString(byte[])}
36   * 
37   * <table>
38   *  <tr>
39   *      <th>Pros</th>
40   *      <th>Cons</th>
41   *  </tr>
42   *  <tr>
43   *      <td>
44   *          Better assurance than {@link HashCodeCacheKeyGenerator} against key collisions though
45   *          with the argument hashCodes being the source of the data it is still not perfect.
46   *      </td>
47   *      <td>
48   *          Slower than {@link HashCodeCacheKeyGenerator}
49   *      </td>
50   *  </tr>
51   * </table>
52   * 
53   * @author Eric Dalquist
54   * @version $Revision$
55   */
56  public class MessageDigestCacheKeyGenerator extends AbstractCacheKeyGenerator<String> {

102     @Override
103     public String generateKey(Object... data) {
104         final MessageDigest messageDigest = this.getMessageDigest();
105         
106         final MessageDigestOutputStream messageDigestOutputStream = new MessageDigestOutputStream(messageDigest);
107         final DataOutputStream dataOutputStream = new DataOutputStream(messageDigestOutputStream);
108         
109         try {
110             this.deepDigest(dataOutputStream, data);
111         }
112         catch (IOException ioe) {
113             throw new IllegalStateException("IOExceptions should not be possible from this key generator");
114         }
115 
116         final byte[] digest = messageDigest.digest();
117         return this.encodeHash(digest);
118     }
119 
120     /**
121      * Encode the digested hash bytes as a String
122      */
123     protected String encodeHash(final byte[] digest) {
124         return Base64.encodeBase64URLSafeString(digest);
125     }

127     /**
128      * Tries to clone the {@link MessageDigest} that was created during construction. If the clone fails
129      * that is remembered and from that point on new {@link MessageDigest} instances will be created on
130      * every call.
131      * 
132      * @return Generates a {@link MessageDigest} to use during a call to {@link #generateKey(Object...)}
133      */
134     protected MessageDigest getMessageDigest() {
135         if (this.cloneNotSupported) {
136             final String algorithm = this.baseMessageDigest.getAlgorithm();
137             try {
138                 return MessageDigest.getInstance(algorithm);
139             }
140             catch (NoSuchAlgorithmException e) {
141                 throw new IllegalStateException("MessageDigest algorithm '" + algorithm + "' was supported when " + this.getClass().getSimpleName() + " was created but is not now. This should not be possible.", e);
142             }
143         }
144         
145         try {
146             return (MessageDigest)this.baseMessageDigest.clone();
147         }
148         catch (CloneNotSupportedException e) {
149             this.cloneNotSupported = true;
150             this.logger.warn("Could not clone MessageDigest using algorithm '" + this.baseMessageDigest.getAlgorithm() + "'. MessageDigest.getInstance will be used from now on which will be much more expensive.", e);
151             return this.getMessageDigest();
152         }
153     }

155     /**
156      * Does a deep traversal, iterating over arrays, collections and maps and map entries. When
157      * a leaf in the object graph is found the appropriate digest method is called.
158      * 
159      * If {@link #setCheckforCycles(boolean)} is true and a cycle is found nothing is appended
160      * to the {@link MessageDigest} the second time the object is encountered.
161      * 
162      * @param o The object to inspect
163      */
164     protected final void deepDigest(DataOutputStream dataOutputStream, Object o) throws IOException {
165         if (o == null || !register(o)) {
166             dataOutputStream.write(0);
167             return;
168         }
169         
170         try {
171             if (o instanceof Class<?>) {
172                 this.digest(dataOutputStream, (Class<?>) o);
173             }
174             else if (o instanceof Enum<?>) {
175                 this.digest(dataOutputStream, (Enum<?>) o);
176             }
177             else if (o.getClass().isArray()) {
178                 final int length = Array.getLength(o);
179                 for (int index = 0; index < length; index++) {
180                     final Object arrayValue = Array.get(o, index);
181                     this.deepDigest(dataOutputStream, arrayValue);
182                 }
183             }
184             else if (o instanceof Iterable<?>) {
185                 for (final Object e : ((Iterable<?>) o)) {
186                     this.deepDigest(dataOutputStream, e);
187                 }
188             }
189             else if (o instanceof Map<?, ?>) {
190                 this.deepDigest(dataOutputStream, ((Map<?, ?>) o).entrySet());
191             }
192             else if (o instanceof Map.Entry<?, ?>) {
193                 final Map.Entry<?, ?> entry = (Map.Entry<?, ?>) o;
194                 this.deepDigest(dataOutputStream, entry.getKey());
195                 this.deepDigest(dataOutputStream, entry.getValue());
196             }
197             else if (o instanceof String) {
198                 dataOutputStream.writeUTF((String)o);
199             }
200             else if (o instanceof Boolean) {
201                 dataOutputStream.writeBoolean(((Boolean)o).booleanValue());
202             }
203             else if (o instanceof Byte) {
204                 dataOutputStream.writeByte(((Byte)o).byteValue());
205             }
206             else if (o instanceof Character) {
207                 dataOutputStream.writeChar(((Character)o).charValue());
208             }
209             else if (o instanceof Double) {
210                 dataOutputStream.writeDouble(((Double)o).doubleValue());
211             }
212             else if (o instanceof Float) {
213                 dataOutputStream.writeFloat(((Float)o).floatValue());
214             }
215             else if (o instanceof Integer) {
216                 dataOutputStream.writeInt(((Integer)o).intValue());
217             }
218             else if (o instanceof Long) {
219                 dataOutputStream.writeLong(((Long)o).longValue());
220             }
221             else if (o instanceof Short) {
222                 dataOutputStream.writeShort(((Short)o).shortValue());
223             }
224             else {
225                 this.digest(dataOutputStream, o);
226             }
227         }
228         finally {
229             unregister(o);
230         }
231     }

233     /**
234      * Special handling for digesting a {@link Class} which does not implement hashCode.
235      * 
236      * Default implementation is to digest {@link Class#getName()}
237      */
238     protected void digest(DataOutputStream dataOutputStream, Class<?> c) throws IOException {
239         this.digest(dataOutputStream, c.getName());
240     }
241     
242     /**
243      * Generate hash code for an Enum, uses a combination of the Class and name to generate a consistent hash code
244      */
245     protected void digest(DataOutputStream dataOutputStream, Enum<?> e) throws IOException {
246         this.deepDigest(dataOutputStream, new Object[] { e.getClass(), e.name() });
247     }
248 
249     /**
250      * Add an object to the digest. Default adds the object's hashCode to the digest
251      */
252     protected void digest(DataOutputStream dataOutputStream, Object o) throws IOException {
253         dataOutputStream.writeInt(o.hashCode());
254     }
