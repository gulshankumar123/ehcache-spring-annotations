145     public void afterPropertiesSet() throws Exception {
146         if (this.cacheResolverFactory == null) {
147             final DefaultCacheResolverFactory defaultCacheResolverFactory = new DefaultCacheResolverFactory(this.getCacheManager());
148             if (this.createCaches != null) {
149                 defaultCacheResolverFactory.setCreateCaches(this.createCaches);
150             }
151             if (this.selfPopulatingCacheScope != null) {
152                 defaultCacheResolverFactory.setSelfPopulatingCacheScope(this.selfPopulatingCacheScope);
153             }
154             defaultCacheResolverFactory.setExecutor(executor);
155             defaultCacheResolverFactory.setScheduler(scheduler);            
156             this.cacheResolverFactory = defaultCacheResolverFactory;
157         }
158         else {
159             if (this.createCaches != null) {
160                 this.logger.warn("createCaches was specified but a custom CacheResolverFactory was also configured. The createCaches value will be ignored.");
161             }
162             if (this.selfPopulatingCacheScope != null) {
163                 this.logger.warn("selfPopulatingCacheScope was specified but a custom CacheResolverFactory was also configured. The selfPopulatingCacheScope value will be ignored.");
164             }
165             if (this.executor != null) {
166                 this.logger.warn("executor was specified but a custom CacheResolverFactory was also configured. The executor value will be ignored.");
167             }
168             if (this.scheduler != null) {
169                 this.logger.warn("scheduler was specified but a custom CacheResolverFactory was also configured. The scheduler value will be ignored.");
170             }
171         }

210     private MethodAttribute getMethodAttribute(final Method method, final Class<?> targetClass) {
211         //Check if the method has already been inspected and should be ignored
212         if (this.ingoredMethods.containsKey(targetClass, method)) {
213             return null;
214         }
215         
216         //Check the cache if the method has already had its advise attributes created
217         final MethodAttribute attributes = this.attributesCache.get(targetClass, method);
218         if (attributes != null) {
219             return attributes;
220         }
221         
222         // We need to work it out.
223         final MethodAttribute att = this.computeMethodAttribute(method, targetClass);
224         
225         // Put it in the cache.
226         if (att == null) {
227             this.ingoredMethods.put(targetClass, method, Boolean.TRUE);
228         }
229         else  {
230             this.logger.debug("Adding {} advised method '{}' with attribute: {}", new Object[] { att.getAdviceType(), method.getName(), att });
231             final MethodAttribute existing = this.attributesCache.putIfAbsent(targetClass, method, att);
232             if (existing != null) {
233                 return existing;
234             }
235         }
236         
237         return att;
238     }

256     private MethodAttribute computeMethodAttribute(Method method, Class<?> targetClass) {
257         // Don't allow no-public methods as required.
258         if (this.allowPublicMethodsOnly() && !Modifier.isPublic(method.getModifiers())) {
259             return null;
260         }
261 
262         // The method may be on an interface, but we need attributes from the target class.
263         // If the target class is null, the method will be unchanged.
264         Method specificMethod = ClassUtils.getMostSpecificMethod(method, targetClass);
265         // If we are dealing with method with generic parameters, find the original method.
266         specificMethod = BridgeMethodResolver.findBridgedMethod(specificMethod);
267 
268         // First try is the method in the target class.
269         MethodAttribute att = this.findMethodAttribute(specificMethod);
270         if (att != null) {
271             return att;
272         }
273 
274         if (specificMethod != method) {
275             // Fallback is to look at the original method.
276             att = this.findMethodAttribute(method);
277             if (att != null) {
278                 return att;
279             }
280         }
281         
282         return null;
283     }

285     /**
286      * Determine if the specified {@link AnnotatedElement} is annotated with either {@link Cacheable} or {@link TriggersRemove}
287      * 
288      * @param method The element to inspect
289      * @return The advice attributes about the element, null if the element is not advised
290      */
291     private MethodAttribute findMethodAttribute(Method method) {
292         Cacheable cacheableAnnotation = method.getAnnotation(Cacheable.class);
293         if (cacheableAnnotation != null) {
294             final ParameterMask parameterMask = this.parsePartialCacheKeyAnnotations(method);
295             return this.parseCacheableAnnotation(cacheableAnnotation, method, parameterMask);
296         }
297         
298         TriggersRemove triggersRemove = method.getAnnotation(TriggersRemove.class);
299         if (triggersRemove != null) {
300             final ParameterMask parameterMask = this.parsePartialCacheKeyAnnotations(method);
301             return this.parseTriggersRemoveAnnotation(triggersRemove, method, parameterMask);
302         }
303         
304         for (final Annotation metaAnn : method.getAnnotations()) {
305             final Class<? extends Annotation> annotationType = metaAnn.annotationType();
306             cacheableAnnotation = annotationType.getAnnotation(Cacheable.class);
307             if (cacheableAnnotation != null) {
308                 final ParameterMask parameterMask = this.parsePartialCacheKeyAnnotations(method);
309                 return this.parseCacheableAnnotation(cacheableAnnotation, method, parameterMask);
310             }
311             
312             triggersRemove = annotationType.getAnnotation(TriggersRemove.class);
313             if (triggersRemove != null) {
314                 final ParameterMask parameterMask = this.parsePartialCacheKeyAnnotations(method);
315                 return this.parseTriggersRemoveAnnotation(triggersRemove, method, parameterMask);
316             }
317         }
318 
319         return null;
320     }

322     /**
323      * Parse the parameters annotated with {@link PartialCacheKey}.
324      * 
325      * @param method The method who's parameters should be checked.
326      * @return The parameter mask for the annotated parameters.
327      */
328     protected ParameterMask parsePartialCacheKeyAnnotations(Method method) {
329         final Annotation[][] parameterAnnotations = method.getParameterAnnotations();
330         final boolean[] mask = new boolean[parameterAnnotations.length];
331         for (int i = 0; i < parameterAnnotations.length; i++) {
332             final Annotation[] annotations = parameterAnnotations[i];
333             for (final Annotation annotation : annotations) {
334                 final Class<? extends Annotation> annotationType = annotation.annotationType();
335                 if (PartialCacheKey.class.equals(annotationType)) {
336                     mask[i] = true;
337                     break;
338                 }
339             }
340         }
341         
342         return ParameterMask.create(mask);
343     }

345     /**
346      * Construct a {@link CacheableAttribute} from a {@link Cacheable} annotation.
347      * 
348      * @param ann The annotation to build the attributes from
349      * @return The constructed cacheable advise attributes
350      */
351     protected CacheableAttribute parseCacheableAnnotation(Cacheable ann, Method method, ParameterMask parameterMask) {
352         final String resolverFactoryName = ann.resolverFactoryName();
353         final ResolverFactory resolverFactory = ann.resolverFactory();
354         final CacheResolverFactory cacheResolverFactory = this.getCacheResolverFactory(resolverFactoryName, resolverFactory);
355         
356         final CacheableCacheResolver cacheResolver = cacheResolverFactory.getCacheResolver(ann, method);
357         
358         final String keyGeneratorName = ann.keyGeneratorName();
359         final KeyGenerator keyGenerator = ann.keyGenerator();
360         final CacheKeyGenerator<? extends Serializable> cacheKeyGenerator = this.getCacheKeyGenerator(keyGeneratorName, keyGenerator);
361         
362         final String cacheableInteceptorName = ann.cacheableInteceptorName();
363         final CacheableInterceptor cacheInterceptor = this.getCacheInterceptor(cacheableInteceptorName);
364         
365         final boolean cacheNull = ann.cacheNull();
366         final DecoratedCacheType decoratedCacheType = DecoratedCacheType.getDecoratedCacheType(ann, method);
367         if (!cacheNull && (decoratedCacheType == DecoratedCacheType.REFRESHING_SELF_POPULATING_CACHE || decoratedCacheType == DecoratedCacheType.SELF_POPULATING_CACHE)) {
368             this.logger.warn("cacheNull is set to false and decoratedCacheType is " + decoratedCacheType + ", cacheNull will be ignored on: " + method);
369         }
370 
371         return new CacheableAttributeImpl(cacheResolver, cacheKeyGenerator, parameterMask, cacheNull, cacheInterceptor);
372     }

488     protected final <T> T getOrCreateChildBean(Class<T> beanType, String beanClass, Property[] properties) {
489         final StringBuilder beanNameBuilder = new StringBuilder();
490         
491         beanNameBuilder.append(beanClass);
492         
493         final MutablePropertyValues mutablePropertyValues = new MutablePropertyValues();
494         
495         //Sort the properties array first so bean name generation is always consistent
496         Arrays.sort(properties, PropertyComparator.INSTANCE);
497         
498         for (Property property : properties) {
499             final String name = property.name();
500             final String value = property.value();
501             final String ref = property.ref();
502             
503             beanNameBuilder.append("[").append(name).append(",").append(value).append(",").append(ref).append("]");
504             
505             if (value.length() > 0) {
506                 if (ref.length() > 0) {
507                     throw new IllegalArgumentException("Only one of value or ref must be specified no both on Property with name: " + name);
508                 }
509                 
510                 mutablePropertyValues.addPropertyValue(name, value);
511             }
512             else if (ref.length() > 0) {
513                 mutablePropertyValues.addPropertyValue(name, new RuntimeBeanReference(ref));
514             }
515             else {
516                 throw new IllegalArgumentException("Either value or ref must be specified on Property with name: " + name);
517             }
518         }
519         
520         final String beanName = beanNameBuilder.toString();
521         
522         //See if the bean is already registered using the compiled bean name, if so just use that instance
523         if (this.childBeanFactory.containsBean(beanName)) {
524             return this.childBeanFactory.getBean(beanName, beanType);
525         }
526 
527         //Create and register the bean if it didn't already exist
528         final AbstractBeanDefinition beanDefinition;
529         try {
530             beanDefinition = BeanDefinitionReaderUtils.createBeanDefinition(null, beanClass, ClassUtils.getDefaultClassLoader());
531         }
532         catch (ClassNotFoundException e) {
533             throw new BeanCreationException("Could not find class '" + beanClass + "' to create " + beanType + " from", e);
534         }
535         
536         if (ReflectionHelperAware.class.isAssignableFrom(beanDefinition.getBeanClass())) {
537             mutablePropertyValues.addPropertyValue("reflectionHelper", this.reflectionHelper);
538         }
539         
540         beanDefinition.setPropertyValues(mutablePropertyValues);
541         this.childBeanFactory.registerBeanDefinition(beanName, beanDefinition);
542         
543         return this.childBeanFactory.getBean(beanName, beanType);
544     }
