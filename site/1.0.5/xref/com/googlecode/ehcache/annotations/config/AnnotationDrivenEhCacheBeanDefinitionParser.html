53  /**
54   * {@link BeanDefinitionParser} that sets up {@link DefaultBeanFactoryPointcutAdvisor}
55   * instances to wrap {@link EhCacheInterceptor}s around {@link Cacheable}s and {@link TriggersRemove}
56   * advised methods.
57   * 
58   * @author Nicholas Blair, nblair@doit.wisc.edu
59   * @version $Id$
60   */
61  public class AnnotationDrivenEhCacheBeanDefinitionParser implements BeanDefinitionParser {
62      public static final String XSD_ATTR__CREATE_MISSING_CACHES = "create-missing-caches";
63      public static final String XSD_ATTR__CACHE_MANAGER = "cache-manager";
64      public static final String XSD_ATTR__DEFAULT_CACHE_KEY_GENERATOR = "default-cache-key-generator";
65      public static final String XSD_ATTR__SELF_POPULATING_CACHE_SCOPE = "self-populating-cache-scope";
66  
67      public static final String EHCACHE_CACHING_ADVISOR_BEAN_NAME = AnnotationDrivenEhCacheBeanDefinitionParser.class.getPackage().getName() + ".internalEhCacheCachingAdvisor";
68      
69      public static final String DEFAULT_CACHE_KEY_GENERATOR = HashCodeCacheKeyGenerator.DEFAULT_BEAN_NAME;  
70      
71      
72      /* (non-Javadoc)
73       * @see org.springframework.beans.factory.xml.BeanDefinitionParser#parse(org.w3c.dom.Element, org.springframework.beans.factory.xml.ParserContext)
74       */
75      public BeanDefinition parse(Element element, ParserContext parserContext) {
76          AopNamespaceUtils.registerAutoProxyCreatorIfNecessary(parserContext, element);
77          if (!parserContext.getRegistry().containsBeanDefinition(EHCACHE_CACHING_ADVISOR_BEAN_NAME)) {
78              final Object elementSource = parserContext.extractSource(element);
79              
80              final RuntimeBeanReference defaultCacheKeyGeneratorReference = 
81                  this.setupDefaultCacheKeyGenerators(element, parserContext, elementSource);
82              
83              final RuntimeBeanReference cacheAttributeSourceReference = 
84                  this.setupCacheAttributeSource(element, parserContext, elementSource, defaultCacheKeyGeneratorReference);
85              
86              final RuntimeBeanReference pointcutReference = 
87                  this.setupPointcut(element, parserContext, elementSource, cacheAttributeSourceReference);
88              
89              final RuntimeBeanReference interceptorReference = 
90                  this.setupInterceptor(element, parserContext, elementSource, cacheAttributeSourceReference);
91              
92              this.setupPointcutAdvisor(element, parserContext, elementSource, pointcutReference, interceptorReference);
93             
94          }
95          return null;
96      }

98      /**
99       * Setup the default cache key generator. 
100      * 
101      * @return A reference to the default cache key generator. Should never be null.
102      */
103     protected RuntimeBeanReference setupDefaultCacheKeyGenerators(Element element, ParserContext parserContext, Object elementSource) {
104         //Register all of the default cache key generator types
105         this.setupDefaultCacheKeyGenerator(ListCacheKeyGenerator.class, parserContext, elementSource);
106         this.setupDefaultCacheKeyGenerator(HashCodeCacheKeyGenerator.class, parserContext, elementSource);
107         this.setupDefaultCacheKeyGenerator(MessageDigestCacheKeyGenerator.class, parserContext, elementSource);
108         this.setupDefaultCacheKeyGenerator(ReflectionHashCodeCacheKeyGenerator.class, parserContext, elementSource);
109         this.setupDefaultCacheKeyGenerator(StringCacheKeyGenerator.class, parserContext, elementSource);
110         
111         //If the default cache key generator was specified simply return a bean reference for that
112         final String defaultCacheKeyGeneratorName = element.getAttribute(XSD_ATTR__DEFAULT_CACHE_KEY_GENERATOR);
113         if (StringUtils.hasLength(defaultCacheKeyGeneratorName)) {
114             return new RuntimeBeanReference(defaultCacheKeyGeneratorName);
115         }
116         
117         //Use the default name for the bean reference
118         return new RuntimeBeanReference(DEFAULT_CACHE_KEY_GENERATOR);
119     }

121     /**
122      * Utility API used to setup each of the default {@link CacheKeyGenerator} implementations. Requires
123      * that the class has a static String field named DEFAULT_BEAN_NAME declared that is used for the bean
124      * name. 
125      */
126     protected final void setupDefaultCacheKeyGenerator(Class<? extends CacheKeyGenerator<? extends Serializable>> generatorClass, ParserContext parserContext, Object elementSource) {
127         final String generatorName;
128         try {
129             final Field field = generatorClass.getField("DEFAULT_BEAN_NAME");
130             generatorName = (String)field.get(null);
131         }
132         catch (Exception e) {
133             throw new IllegalArgumentException("Could not access static field 'DEFAULT_BEAN_NAME' on " + generatorClass + ". This field is required to be setup as a default CacheKeyGenerator", e);
134         }
135         
136         final RootBeanDefinition defaultKeyGenerator = new RootBeanDefinition(generatorClass);
137         defaultKeyGenerator.setSource(elementSource);
138         defaultKeyGenerator.setRole(BeanDefinition.ROLE_INFRASTRUCTURE);
139         
140         final BeanDefinitionRegistry registry = parserContext.getRegistry();
141         registry.registerBeanDefinition(generatorName, defaultKeyGenerator);
142     }

145     /**
146      * Create a {@link CacheAttributeSource} bean that will be used by the advisor and interceptor
147      * 
148      * @return Reference to the {@link CacheAttributeSource}. Should never be null.
149      */
150     protected RuntimeBeanReference setupCacheAttributeSource(Element element, ParserContext parserContext, Object elementSource, RuntimeBeanReference defaultCacheKeyGenerator) {
151         final RootBeanDefinition cacheAttributeSource = new RootBeanDefinition(CacheAttributeSourceImpl.class);
152         cacheAttributeSource.setSource(elementSource);
153         cacheAttributeSource.setRole(BeanDefinition.ROLE_INFRASTRUCTURE);
154         
155         final MutablePropertyValues propertyValues = cacheAttributeSource.getPropertyValues();
156         RuntimeBeanReference cacheManagerReference = new RuntimeBeanReference(element.getAttribute(XSD_ATTR__CACHE_MANAGER));
157         propertyValues.addPropertyValue("cacheManager", cacheManagerReference);
158         propertyValues.addPropertyValue("createCaches", Boolean.parseBoolean(element.getAttribute(XSD_ATTR__CREATE_MISSING_CACHES)));
159         propertyValues.addPropertyValue("defaultCacheKeyGenerator", defaultCacheKeyGenerator);
160         final String blockingCacheScope = element.getAttribute(XSD_ATTR__SELF_POPULATING_CACHE_SCOPE);
161         if (blockingCacheScope != null) {
162             propertyValues.addPropertyValue("selfPopulatingCacheScope", SelfPopulatingCacheScope.valueOf(blockingCacheScope.toUpperCase()));
163         }
164 
165         final XmlReaderContext readerContext = parserContext.getReaderContext();
166         final String cacheAttributeSourceBeanName = readerContext.registerWithGeneratedName(cacheAttributeSource);
167         return new RuntimeBeanReference(cacheAttributeSourceBeanName);
168     }

170     /**
171      * Create the {@link Pointcut} used to apply the caching interceptor
172      * 
173      * @return Reference to the {@link Pointcut}. Should never be null.
174      */
175     protected RuntimeBeanReference setupPointcut(Element element, ParserContext parserContext, Object elementSource, RuntimeBeanReference cacheAttributeSource) {
176         final RootBeanDefinition pointcut = new RootBeanDefinition(CacheStaticMethodMatcherPointcut.class);
177         pointcut.setSource(elementSource);
178         pointcut.setRole(BeanDefinition.ROLE_INFRASTRUCTURE);
179         
180         final MutablePropertyValues propertyValues = pointcut.getPropertyValues();
181         propertyValues.addPropertyValue("cacheAttributeSource", cacheAttributeSource);
182         
183         final XmlReaderContext readerContext = parserContext.getReaderContext();
184         final String pointcutBeanName = readerContext.registerWithGeneratedName(pointcut);
185         return new RuntimeBeanReference(pointcutBeanName);
186     }

188     /**
189      * Create {@link MethodInterceptor} that is applies the caching logic to advised methods.
190      * 
191      * @return Reference to the {@link MethodInterceptor}. Should never be null.
192      */
193     protected RuntimeBeanReference setupInterceptor(Element element, ParserContext parserContext, Object elementSource, RuntimeBeanReference cacheableAttributeSourceRuntimeReference) {
194         final RootBeanDefinition interceptor = new RootBeanDefinition(EhCacheInterceptor.class);
195         interceptor.setSource(elementSource);
196         interceptor.setRole(BeanDefinition.ROLE_INFRASTRUCTURE);
197         
198         final MutablePropertyValues propertyValues = interceptor.getPropertyValues();
199         propertyValues.addPropertyValue("cacheAttributeSource", cacheableAttributeSourceRuntimeReference);
200         
201         final XmlReaderContext readerContext = parserContext.getReaderContext();
202         final String interceptorBeanName = readerContext.registerWithGeneratedName(interceptor);
203         return new RuntimeBeanReference(interceptorBeanName);
204     }

206     /**
207      * Create {@link PointcutAdvisor} that puts the {@link Pointcut} and {@link MethodInterceptor} together.
208      * 
209      * @return Reference to the {@link PointcutAdvisor}. Should never be null.
210      */
211     protected RuntimeBeanReference setupPointcutAdvisor(Element element, ParserContext parserContext, Object elementSource, RuntimeBeanReference cacheablePointcutBeanReference,  RuntimeBeanReference cachingInterceptorBeanReference) {
212         final RootBeanDefinition pointcutAdvisor = new RootBeanDefinition(DefaultBeanFactoryPointcutAdvisor.class);
213         pointcutAdvisor.setSource(elementSource);
214         pointcutAdvisor.setRole(BeanDefinition.ROLE_INFRASTRUCTURE);
215 
216         final MutablePropertyValues propertyValues = pointcutAdvisor.getPropertyValues();
217         propertyValues.addPropertyValue("adviceBeanName", cachingInterceptorBeanReference.getBeanName());
218         propertyValues.addPropertyValue("pointcut", cacheablePointcutBeanReference);
219         if (element.hasAttribute("order")) {
220             propertyValues.addPropertyValue("order", element.getAttribute("order"));
221         }
222         
223         final BeanDefinitionRegistry registry = parserContext.getRegistry();
224         registry.registerBeanDefinition(EHCACHE_CACHING_ADVISOR_BEAN_NAME, pointcutAdvisor);
225         return new RuntimeBeanReference(EHCACHE_CACHING_ADVISOR_BEAN_NAME);
226     }
