167     @Test
168     public void testCacheKeyGeneratorPerformance() throws NoSuchAlgorithmException, InterruptedException, BrokenBarrierException {
169         final Map<String, AbstractDeepCacheKeyGenerator<?, ? extends Serializable>> generators = new LinkedHashMap<String, AbstractDeepCacheKeyGenerator<?, ? extends Serializable>>();
170         
171         generators.put("ListCacheKeyGenerator", new ListCacheKeyGenerator());
172         generators.put("StringCacheKeyGenerator", new StringCacheKeyGenerator());
173         generators.put("MessageDigestCacheKeyGenerator_\"MD5\"", new MessageDigestCacheKeyGenerator("MD5"));
174         generators.put("HashCodeCacheKeyGenerator", new HashCodeCacheKeyGenerator());
175         
176         final ThreadGroupRunner threadGroupRunner = new ThreadGroupRunner("CacheKeyGeneratorPerformanceTest-", true);
177 
178         final int threads = 20;
179         this.testThreadStateLatch = new CyclicBarrier(threads + 1);
180         threadGroupRunner.addTask(threads, new KeyGenerationRunnable());
181         threadGroupRunner.start();
182         
183         
184         for (int totalLoopCount = 2; totalLoopCount <= 8; totalLoopCount*=2) {
185             final long duration = 1000 * totalLoopCount;
186             System.out.println("Sleeping 5s Before: " + duration);
187             Thread.sleep(5*1000);
188             
189             for (final Map.Entry<String, AbstractDeepCacheKeyGenerator<?, ? extends Serializable>> generatorEntry : generators.entrySet()) {
190                 
191                 for (int configIndex = 0; configIndex < 3; configIndex++) {
192                     //Setup state and start threads
193                     this.totalKeyCount.set(0);
194                     this.generator = generatorEntry.getValue();
195                     this.runTest = true;
196                     this.testThreadStateLatch.await();
197                     
198                     final String generatorConfig;
199                     switch (configIndex) {
200                         case 0:
201                             this.generator.setCheckforCycles(false);
202                             this.generator.setIncludeMethod(false);
203                             this.generator.setIncludeParameterTypes(false);
204                             this.generator.setUseReflection(false);
205                             generatorConfig = ",setIncludeMethod(false),setIncludeParameterTypes(false),setUseReflection(false)";
206                         break;
207                         
208                         case 1:
209                             this.generator.setCheckforCycles(false);
210                             this.generator.setIncludeMethod(true);
211                             this.generator.setIncludeParameterTypes(true);
212                             this.generator.setUseReflection(false);
213                             generatorConfig = ",setIncludeMethod(true),setIncludeParameterTypes(true),setUseReflection(false)";
214                         break;
215                         
216                         case 2:
217                             AbstractDeepCacheKeyGenerator.IMPLEMENTS_CACHE.clear();
218                             this.generator.setCheckforCycles(false);
219                             this.generator.setIncludeMethod(true);
220                             this.generator.setIncludeParameterTypes(true);
221                             this.generator.setUseReflection(true);
222                             generatorConfig = ",setIncludeMethod(true),setIncludeParameterTypes(true),setUseReflection(true)";
223                         break;
224                         
225                         default:
226                             throw new IllegalStateException();
227                     }
228                     
229                     //Sleep main thread for duration of test
230                     Thread.sleep(duration);
231                     
232                     //Switch the test running flag and wait for threads to update the key count
233                     this.runTest = false;
234                     this.testThreadStateLatch.await();
235                     
236                     final long keyCount = this.totalKeyCount.get();
237                     System.out.println(keyCount + "|" + duration + "|" + keyCount/((double)duration) + "|" + generatorEntry.getKey() + generatorConfig);
238                 }
239             }
240         }
241         
242         synchronized (this.testCompleteMutex) {
243             this.testThreadStateLatch.await();
244             this.testComplete = true;
245         }
246         
247         threadGroupRunner.join();
248     }

250     //Class scoped variables used to control the test threads
251     private final AtomicLong totalKeyCount = new AtomicLong();
252     private AbstractDeepCacheKeyGenerator<?, ? extends Serializable> generator;
253     private CyclicBarrier testThreadStateLatch;
254     private volatile boolean runTest = false;
255     
256     private final Object testCompleteMutex = new Object();
257     private boolean testComplete = false;
258     
259     private class KeyGenerationRunnable implements Runnable {
260         public void run() {
261             while (!testComplete) {
262                 await();
263                 synchronized (testCompleteMutex) {
264                     if (testComplete) {
265                         testCompleteMutex.notifyAll();
266                         return;
267                     }
268                 }
269                 
270                 long keyCount = 0;
271                 while (runTest) {
272                     Serializable lastKey = null;
273                     for (final MethodInvocation invocation : invocations) {
274                         final Serializable key = generator.generateKey(invocation);
275                         //include cost of generated keys hashCode API
276                         for (int index = 0; index < 10; index++) {
277                             key.hashCode();
278                         }
279                         key.equals(lastKey);
280                         lastKey = key;
281                         keyCount++;
282                     }
283                 }
284              
285                 totalKeyCount.addAndGet(keyCount);
286                 await();
287             }
288         }
289 
290         private void await() {
291             if (testComplete) {
292                 return;
293             }
294             try {
295                 testThreadStateLatch.await();
296             }
297             catch (InterruptedException e) {
298                 throw new RuntimeException(e);
299             }
300             catch (BrokenBarrierException e) {
301                 throw new RuntimeException(e);
302             }
303         }
304     }
