28  /**
29   * {@link ReflectionHelper} that uses a cache to track reflection operations. A background thread is started on first
30   * usage to perform cache cleanup to avoid class reference leaks. Code that uses this class should call {@link #destroy()}
31   * when complete to ensure the cleanup thread shuts down.  
32   * 
33   * @author Eric Dalquist
34   * @version $Revision$
35   */
36  public final class CachingReflectionHelper implements DisposableBean, ReflectionHelper {
37      protected final Logger logger = LoggerFactory.getLogger(this.getClass());
38      
39      private enum ImplementsMethod {
40          HASH_CODE,
41          EQUALS,
42          TO_STRING;
43      }
44      
45      private final ReferenceQueue<Object> referenceQueue = new ReferenceQueue<Object>();
46      private final Map<Class<?>, Set<ImplementsMethod>> implementsCache = new ReferenceMap<Class<?>, Set<ImplementsMethod>>(ReferenceType.WEAK, ReferenceType.STRONG, this.referenceQueue);
47      private final Thread cleanupThread;
48      private final AtomicBoolean running = new AtomicBoolean(false);
49  
50      public CachingReflectionHelper() {
51          this.cleanupThread = new Thread("CachingReflectionHelper_CleanupThread") {
52              @Override
53              public void run() {
54                  while (running.get()) {
55                      try {
56                          final Reference<? extends Object> reference = referenceQueue.remove();
57                          ((FinalizableReference) reference).finalizeReferent();
58                      }
59                      catch (InterruptedException e) {
60                          /* ignore */
61                      }
62                      catch (Throwable t) {
63                          logger.error("Error finalizing reference.", t);
64                      }
65                  }
66              }
67          };
68  
69          this.cleanupThread.setDaemon(true);
70      }
71      
72  
73      public void setThreadName(String threadName) {
74          this.cleanupThread.setName("CachingReflectionHelper_CleanupThread" + (threadName != null ? "-" + threadName : ""));
75      }
76  
77  
78  
79      private Map<Class<?>, Set<ImplementsMethod>> getCache() {
80          if (!this.running.get() && this.running.compareAndSet(false, true)) {
81              //Don't start the cleanup thread until the cache is actually accessed
82              this.cleanupThread.start();
83          }
84          
85          return this.implementsCache;
86      }
87  
88      @Override
89      protected void finalize() throws Throwable {
90          this.destroy();
91      }
92  
93      public void destroy() throws Exception {
94          if (this.running.get() && this.running.compareAndSet(true, false)) {
95              this.cleanupThread.interrupt();
96              try {
97                  this.cleanupThread.join(1000);
98              }
99              catch (InterruptedException e) {
100                 logger.warn("Failed to join the " + this.cleanupThread.getName() + " thread");
101             }
102         }
103     }

132     /**
133      * Scans a class to see if it implements the hashCode, toString and equals methods which are commonly
134      * used by key generators
135      */
136     private boolean doesImplement(final Class<?> elementClass, ImplementsMethod method) {
137         final Map<Class<?>, Set<ImplementsMethod>> cache = this.getCache();
138         Set<ImplementsMethod> methodCache = cache.get(elementClass);
139 
140         if (methodCache == null) {
141             methodCache = EnumSet.noneOf(ImplementsMethod.class);
142             cache.put(elementClass, methodCache);
143 
144             //Create final reference for use by anonymous class
145             final Set<ImplementsMethod> implementsSet = methodCache;
146             ReflectionUtils.doWithMethods(elementClass, new MethodCallback() {
147                 public void doWith(Method method) throws IllegalArgumentException, IllegalAccessException {
148                     if (implementsSet.size() == 3 || method.getDeclaringClass() == Object.class) {
149                         return;
150                     }
151                     
152                     if (ReflectionUtils.isEqualsMethod(method)) {
153                         implementsSet.add(ImplementsMethod.EQUALS);
154                     }
155                     else if (ReflectionUtils.isHashCodeMethod(method)) {
156                         implementsSet.add(ImplementsMethod.HASH_CODE);
157                     }
158                     else if (ReflectionUtils.isToStringMethod(method)) {
159                         implementsSet.add(ImplementsMethod.TO_STRING);
160                     }
161                 }
162             });
163         }
164         
165         return methodCache.contains(method);
166     }
