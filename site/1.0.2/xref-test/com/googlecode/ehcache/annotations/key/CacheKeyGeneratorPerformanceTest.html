81      @Test
82      public void testCacheKeyGeneratorPerformance() throws NoSuchAlgorithmException, InterruptedException, BrokenBarrierException {
83          final Map<String, CacheKeyGenerator<? extends Serializable>> generators = new LinkedHashMap<String, CacheKeyGenerator<? extends Serializable>>();
84          
85          generators.put("ListCacheKeyGenerator|true|true", new ListCacheKeyGenerator(true, true));
86          generators.put("ListCacheKeyGenerator|true|false", new ListCacheKeyGenerator(true, false));
87          generators.put("ListCacheKeyGenerator|false|false", new ListCacheKeyGenerator(false, false));
88          generators.put("StringCacheKeyGenerator|true|true", new StringCacheKeyGenerator(true, true));
89          generators.put("StringCacheKeyGenerator|true|false", new StringCacheKeyGenerator(true, false));
90          generators.put("StringCacheKeyGenerator|false|false", new StringCacheKeyGenerator(false, false));
91          generators.put("MessageDigestCacheKeyGenerator_\"MD5\"|true|true", new MessageDigestCacheKeyGenerator("MD5", true, true));
92          generators.put("MessageDigestCacheKeyGenerator_\"MD5\"|true|false", new MessageDigestCacheKeyGenerator("MD5", true, false));
93          generators.put("MessageDigestCacheKeyGenerator_\"MD5\"|false|false", new MessageDigestCacheKeyGenerator("MD5", false, false));
94          generators.put("HashCodeCacheKeyGenerator|true|true", new HashCodeCacheKeyGenerator(true, true));
95          generators.put("HashCodeCacheKeyGenerator|true|false", new HashCodeCacheKeyGenerator(true, false));
96          generators.put("HashCodeCacheKeyGenerator|false|false", new HashCodeCacheKeyGenerator(false, false));
97          generators.put("ReflectionHashCodeCacheKeyGenerator|true|true", new ReflectionHashCodeCacheKeyGenerator(true, true));
98          generators.put("ReflectionHashCodeCacheKeyGenerator|true|false", new ReflectionHashCodeCacheKeyGenerator(true, false));
99          generators.put("ReflectionHashCodeCacheKeyGenerator|false|false", new ReflectionHashCodeCacheKeyGenerator(false, false));
100         
101         final ThreadGroupRunner threadGroupRunner = new ThreadGroupRunner("CacheKeyGeneratorPerformanceTest-", true);
102 
103         final int threads = 20;
104         this.testThreadStateLatch = new CyclicBarrier(threads + 1);
105         threadGroupRunner.addTask(threads, new KeyGenerationRunnable());
106         threadGroupRunner.start();
107         
108         
109         for (int totalLoopCount = 1; totalLoopCount <= 3; totalLoopCount+=1) {
110             final long duration = 1000 * totalLoopCount;
111             System.out.println("Sleeping 5s Before: " + duration);
112             Thread.sleep(5*1000);
113             
114             for (final Map.Entry<String, CacheKeyGenerator<? extends Serializable>> generatorEntry : generators.entrySet()) {
115                 //Setup state and start threads
116                 this.totalKeyCount.set(0);
117                 this.generator = generatorEntry.getValue();
118                 this.runTest = true;
119                 this.testThreadStateLatch.await();
120                 
121                 //Sleep main thread for duration of test
122                 Thread.sleep(duration);
123                 
124                 //Switch the test running flag and wait for threads to update the key count
125                 this.runTest = false;
126                 this.testThreadStateLatch.await();
127                 
128                 final long keyCount = this.totalKeyCount.get();
129                 System.out.println(keyCount + "|" + duration + "|" + keyCount/((double)duration) + "|" + generatorEntry.getKey());
130             }
131         }
132         
133         synchronized (this.testCompleteMutex) {
134             this.testThreadStateLatch.await();
135             this.testComplete = true;
136         }
137         
138         threadGroupRunner.join();
139     }

141     //Class scoped variables used to control the test threads
142     private final AtomicLong totalKeyCount = new AtomicLong();
143     private CacheKeyGenerator<? extends Serializable> generator;
144     private CyclicBarrier testThreadStateLatch;
145     private volatile boolean runTest = false;
146     
147     private final Object testCompleteMutex = new Object();
148     private boolean testComplete = false;
149     
150     private class KeyGenerationRunnable implements Runnable {
151         public void run() {
152             while (!testComplete) {
153                 await();
154                 synchronized (testCompleteMutex) {
155                     if (testComplete) {
156                         testCompleteMutex.notifyAll();
157                         return;
158                     }
159                 }
160                 
161                 long keyCount = 0;
162                 while (runTest) {
163                     for (final MethodInvocation invocation : invocations) {
164                         final Serializable key = generator.generateKey(invocation);
165                         //include cost of generated keys hashCode API
166                         for (int index = 0; index < 10; index++) {
167                             key.hashCode(); 
168                         }
169                         keyCount++;
170                     }
171                 }
172              
173                 totalKeyCount.addAndGet(keyCount);
174                 await();
175             }
176         }
177 
178         private void await() {
179             if (testComplete) {
180                 return;
181             }
182             try {
183                 testThreadStateLatch.await();
184             }
185             catch (InterruptedException e) {
186                 throw new RuntimeException(e);
187             }
188             catch (BrokenBarrierException e) {
189                 throw new RuntimeException(e);
190             }
191         }
192     }
