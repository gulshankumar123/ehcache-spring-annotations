43  @ContextConfiguration("/triggersRemoveAfterMethodInvocationTestContext.xml")
44  public class TriggersRemoveAfterMethodInvocationTest {
45  
46      private TriggersRemoveAfterMethodInvocationImpl triggersRemoveTestImpl;
47      private CacheManager cacheManager;
48      /**
49       * @param triggersRemoveTestImpl the triggersRemoveTestImpl to set
50       */
51      @Autowired
52      public void setTriggersRemoveTestImpl(
53              TriggersRemoveAfterMethodInvocationImpl triggersRemoveTestImpl) {
54          this.triggersRemoveTestImpl = triggersRemoveTestImpl;
55      }
56      /**
57       * @param cacheManager the cacheManager to set
58       */
59      @Autowired
60      public void setCacheManager(CacheManager cacheManager) {
61          this.cacheManager = cacheManager;
62      }
63  
64      /**
65       * 
66       */
67      @Test
68      public void testHarness() {
69          Cache cache = cacheManager.getCache("triggersRemoveAfterInvocationTestCache");
70          
71          Assert.assertEquals(0, cache.getSize());
72          // place an element in the cache so we can tell when removeAll was called
73          cache.put(new Element("key", "value"));
74          Assert.assertEquals(1, cache.getSize());
75          
76          ThreadGroupRunner groupRunner = new ThreadGroupRunner("triggersRemoveAfterMethodInvocationTest", true);
77          groupRunner.addTask(new Runnable() {
78              public void run() {
79                  //2 
80                  triggersRemoveTestImpl.afterMethodInvocation("foo");
81                  try {
82                      // 6
83                      triggersRemoveTestImpl.getCyclicBarrier().await();
84                  } catch (InterruptedException e) {
85                      Assert.fail(e.getMessage());
86                      e.printStackTrace();
87                  } catch (BrokenBarrierException e) {
88                      Assert.fail(e.getMessage());
89                      e.printStackTrace();
90                  }
91              }
92          });
93          
94          // 1
95          groupRunner.start();
96          try {
97              // 2
98              triggersRemoveTestImpl.getCyclicBarrier().await();
99          } catch (InterruptedException e) {
100             Assert.fail(e.getMessage());
101             e.printStackTrace();
102         } catch (BrokenBarrierException e) {
103             Assert.fail(e.getMessage());
104             e.printStackTrace();
105         }
106         
107         
108         // 4: child thread inside TriggersRemove annotated method
109         Assert.assertEquals(1, cache.getSize());
110         // 5a
111         triggersRemoveTestImpl.getProceedLatch().countDown();
112         // TriggersRemove annotated method invocation complete
113         try {
114             // 6 
115             triggersRemoveTestImpl.getCyclicBarrier().await();
116         } catch (InterruptedException e) {
117             Assert.fail(e.getMessage());
118             e.printStackTrace();
119         } catch (BrokenBarrierException e) {
120             Assert.fail(e.getMessage());
121             e.printStackTrace();
122         }
123         
124         // at this point the removeAll has been called, verify cache is empty
125         Assert.assertEquals(0, cache.getSize());
126         try {
127             groupRunner.join();
128         } catch (InterruptedException e) {
129             e.printStackTrace();
130         }
131     }
