38  /**
39   * {@link BeanDefinitionParser} implementation to process the
40   * "config" element.
41   * 
42   * @author Nicholas Blair
43   * @version $Id$
44   */
45  public final class EhCacheConfigBeanDefinitionParser implements
46  BeanDefinitionParser {
47  
48      public static final String XSD_ELEMENT__EVICT_EXPIRED_ELEMENTS = "evict-expired-elements";
49      public static final String XSD_ATTRIBUTE__INTERVAL = "interval";
50      public static final String XSD_ELEMENT__INCLUDE = "include";
51      public static final String XSD_ELEMENT__EXCLUDE = "exclude";
52      public static final String XSD_ATTRIBUTE__NAME = "name";
53      public static final String XSD_ATTRIBUTE__PATTERN = "pattern";
54  
55      public static final String EHCACHE_CONFIG_EVICTION_TASK_BEAN_NAME = EhCacheConfigBeanDefinitionParser.class.getPackage().getName() + ".internalEhCacheEvictionTask";
56  
57      public static final CacheNameMatcher INCLUDE_ALL_CACHE_NAME_MATCHER = new PatternCacheNameMatcherImpl(".*");
58      /* (non-Javadoc)
59       * @see org.springframework.beans.factory.xml.BeanDefinitionParser#parse(org.w3c.dom.Element, org.springframework.beans.factory.xml.ParserContext)
60       */
61      public BeanDefinition parse(final Element element, final ParserContext parserContext) {
62          final Object elementSource = parserContext.extractSource(element);
63  
64          final NodeList evictExpiredElements = element.getElementsByTagNameNS(element.getNamespaceURI(),XSD_ELEMENT__EVICT_EXPIRED_ELEMENTS);
65          if (evictExpiredElements.getLength() > 1) {
66              throw new BeanCreationException("Only one '" + XSD_ELEMENT__EVICT_EXPIRED_ELEMENTS + "' is allowed");
67          }
68  
69          final int evictExpiredElementsLength = evictExpiredElements.getLength();
70          if (evictExpiredElementsLength == 1) {
71              final Element evictExpiredElement = (Element)evictExpiredElements.item(0);
72  
73              final String interval = evictExpiredElement.getAttribute(XSD_ATTRIBUTE__INTERVAL);
74  
75              List<CacheNameMatcher> cacheNameMatchers = parseEvictExpiredElement(evictExpiredElement);
76  
77              // get RuntimeBeanReference for cacheManager
78              final RuntimeBeanReference cacheManagerReference = new RuntimeBeanReference(element.getAttribute(AnnotationDrivenEhCacheBeanDefinitionParser.XSD_ATTR__CACHE_MANAGER));
79  
80              // make RootBeanDefinition, RuntimeBeanReference for ExpiredElementEvictor instance, wire cacheNameMatchers reference and 
81              final RootBeanDefinition expiredElementEvictor = new RootBeanDefinition(ExpiredElementEvictor.class);
82              expiredElementEvictor.setSource(elementSource);
83              expiredElementEvictor.setRole(BeanDefinition.ROLE_INFRASTRUCTURE);
84  
85              final MutablePropertyValues propertyValues = expiredElementEvictor.getPropertyValues();
86              propertyValues.addPropertyValue("cacheManager", cacheManagerReference);
87              propertyValues.addPropertyValue("cacheNameMatchers", cacheNameMatchers);
88              propertyValues.addPropertyValue("interval", interval);
89  
90              // register expiredElementEvictor
91              final BeanDefinitionRegistry registry = parserContext.getRegistry();
92              registry.registerBeanDefinition(EHCACHE_CONFIG_EVICTION_TASK_BEAN_NAME, expiredElementEvictor);
93          }
94  
95          return null;
96      }
97  
98      /**
99       * @param evictExpiredElement
100      * @return The list of {@link CacheNameMatcher}s to use for finding caches to evict
101      */
102     protected List<CacheNameMatcher> parseEvictExpiredElement(final Element evictExpiredElement) {
103         List<CacheNameMatcher> cacheNameMatchers = new ArrayList<CacheNameMatcher>();
104         final NodeList childNodes = evictExpiredElement.getChildNodes();
105         final int childNodesLength = childNodes.getLength();
106         boolean configContainsExcludes = false;
107         boolean configContainsIncludes = false;
108         if(0 != childNodesLength) {
109             for (int index = 0; index < childNodesLength; index++) {
110                 final Node childNode = childNodes.item(index);
111                 if(Node.ELEMENT_NODE == childNode.getNodeType()) {
112                     final String childName = childNode.getLocalName();
113                     NamedNodeMap childAttributes = childNode.getAttributes();
114                     Node nameAttr = childAttributes.getNamedItem(XSD_ATTRIBUTE__NAME);
115                     CacheNameMatcher matcher = null;
116                     if(null != nameAttr) {
117                         String matcherValue = nameAttr.getTextContent();
118                         matcher = new ExactCacheNameMatcherImpl(matcherValue);
119                     } else {
120                         Node patternAttr = childAttributes.getNamedItem(XSD_ATTRIBUTE__PATTERN);
121                         if(null != patternAttr) {
122                             String matcherValue = patternAttr.getTextContent();
123                             matcher = new PatternCacheNameMatcherImpl(matcherValue);
124                         }
125                     }
126 
127                     if(null != matcher) {
128                         if(XSD_ELEMENT__INCLUDE.equals(childName)) {
129                             cacheNameMatchers.add(matcher);
130                             configContainsIncludes = true;
131                         } else if(XSD_ELEMENT__EXCLUDE.equals(childName)) {
132                             cacheNameMatchers.add(new NotCacheNameMatcherImpl(matcher));
133                             configContainsExcludes = true;
134                         }
135                     }
136                 }
137             }
138         } 
139         
140         if(0 == cacheNameMatchers.size()) {
141             // no include/exclude elements found
142             cacheNameMatchers = Collections.singletonList(INCLUDE_ALL_CACHE_NAME_MATCHER);
143         } else if(!configContainsIncludes && configContainsExcludes) {
144             //config contains excludes only
145             // create a new list with a Include all matcher at the front
146             List<CacheNameMatcher> newList = new ArrayList<CacheNameMatcher>();
147             newList.add(INCLUDE_ALL_CACHE_NAME_MATCHER);
148             newList.addAll(cacheNameMatchers);
149             cacheNameMatchers = newList;
150         }
151         
152         return Collections.unmodifiableList(cacheNameMatchers);
153     }
