17  package com.googlecode.ehcache.annotations.key;
18  
19  import java.io.Serializable;
20  import java.lang.reflect.Method;
21  import java.util.IdentityHashMap;
22  import java.util.Map;
23  
24  import org.aopalliance.intercept.MethodInvocation;

26  /**
27   * Base class for cache key generators. Handles common logic for including/excluding the method
28   * signature from key generation. Also provides support for avoiding circular references for key
29   * generators that traverse the argument object graph via the {@link #register(Object)} and
30   * {@link #unregister(Object)} APIs 
31   * 
32   * @author Eric Dalquist
33   * @version $Revision$
34   */
35  public abstract class AbstractCacheKeyGenerator<T extends Serializable> implements CacheKeyGenerator<T> {
36      //Used by circular reference tracking register/unregister APIs
37      private static final ThreadLocal<Map<Object, Object>> REGISTRY = new ThreadLocal<Map<Object, Object>>() {
38          @Override
39          protected Map<Object, Object> initialValue() {
40              return new IdentityHashMap<Object, Object>();
41          }
42      };
43      
44      private boolean checkforCycles = false;
45      private boolean includeMethod = true;
46      private boolean includeParameterTypes = true;
47      
48      /**
49       * Default constructor, same as calling {@link #AbstractCacheKeyGenerator(boolean, boolean)} with (true, true) 
50       */
51      public AbstractCacheKeyGenerator() {
52          this(true, true);
53      }
54      
55      /**
56       * @see AbstractCacheKeyGenerator#setIncludeMethod(boolean)
57       * @see AbstractCacheKeyGenerator#setIncludeParameterTypes(boolean)
58       */
59      public AbstractCacheKeyGenerator(boolean includeMethod, boolean includeParameterTypes) {
60          this.includeMethod = includeMethod;
61          this.includeParameterTypes = includeParameterTypes;
62      }
63  
64      public final boolean isIncludeMethod() {
65          return includeMethod;
66      }
67      /**
68       * Determines if the invoked method signature should be included in the generated cache key. If
69       * true {@link Method#getDeclaringClass()}, {@link Method#getName()}, and {@link Method#getReturnType()}
70       * are included in the object array the key is generated from. Note that the effect of this option is
71       * such that two methods with the same arguments will have different keys if true. If you have two methods
72       * in the same class with the same name and return value you may want to enable {@link #setIncludeParameterTypes(boolean)}
73       * for even more specific key generation.
74       * 
75       * Note that including the method signature in key generation reduces key generation speed.
76       * 
77       * @param includeMethod true If the {@link Method} from the {@link MethodInvocation} should be
78       *                      included in the generated key, defaults to true.
79       * @see #setIncludeParameterTypes(boolean)
80       */
81      public final void setIncludeMethod(boolean includeMethod) {
82          this.includeMethod = includeMethod;
83      }
84      
85      public final boolean isIncludeParameterTypes() {
86          return includeParameterTypes;
87      }
88      /**
89       * Determines if the method parameter types returned by {@link Method#getParameterTypes()} should be
90       * included in the generated key. This is broken out into a separate option because the call results
91       * in a clone() call on the Class[] every time {@link Method#getParameterTypes()} which reduces key
92       * generation speed.
93       * 
94       * This is option is only used if {@link #setIncludeMethod(boolean)} is true.
95       * 
96       * @param includeParameterTypes true if the {@link Method#getParameterTypes()} should be included in
97       *                              the generated key, defaults to false.
98       * @see #setIncludeMethod(boolean)
99       */
100     public final void setIncludeParameterTypes(boolean includeParameterTypes) {
101         this.includeParameterTypes = includeParameterTypes;
102     }
103     
104     public final boolean isCheckforCycles() {
105         return checkforCycles;
106     }
107     /**
108      * If the key generation should gracefully handle object graph cycles. If false the {@link #register(Object)} and
109      * {@link #unregister(Object)} methods are non-functional. If true the implementation of this class must correctly
110      * utilize the {@link #register(Object)} and {@link #unregister(Object)} methods to track visited objects.
111      * 
112      * @param checkforCycles Defaults to false.
113      */
114     public final void setCheckforCycles(boolean checkforCycles) {
115         this.checkforCycles = checkforCycles;
116     }
117 
118     /* (non-Javadoc)
119      * @see com.googlecode.ehcache.annotations.key.CacheKeyGenerator#generateKey(org.aopalliance.intercept.MethodInvocation)
120      */
121     public final T generateKey(MethodInvocation methodInvocation) {
122         final Object[] arguments = methodInvocation.getArguments();
123         
124         if (this.includeMethod) {
125             final Method method = methodInvocation.getMethod();
126             
127             final Class<?> declaringClass = method.getDeclaringClass();
128             final String name = method.getName();
129             final Class<?> returnType = method.getReturnType();
130 
131             if (this.includeParameterTypes) {
132                 final Class<?>[] parameterTypes = method.getParameterTypes();
133                 return this.generateKey(declaringClass, name, returnType, parameterTypes, arguments);
134             }
135             
136             return this.generateKey(declaringClass, name, returnType, arguments);
137         }
138         
139         try {
140             return this.generateKey(arguments);
141         }
142         finally {
143             if (this.checkforCycles) {
144                 //Cleanup our thread local data
145                 REGISTRY.remove();
146             }
147         }
148     }
149

151     /**
152      * <p>
153      * Registers the given object. Used by the reflection methods to avoid infinite loops.
154      * </p>
155      * 
156      * @param element The object to register.
157      * @return true if the object is not currently registered
158      */
159     protected final boolean register(Object element) {
160         if (!this.checkforCycles) {
161             return true;
162         }
163         
164         final Map<Object, Object> registry = REGISTRY.get();
165         return registry.put(element, element) == null;
166     }
167 
168     /**
169      * <p>
170      * Unregisters the given object. Used by the reflection methods to avoid infinite loops.
171      * </p>
172      */
173     protected final void unregister(Object element) {
174         if (!this.checkforCycles) {
175             return;
176         }
177         
178         final Map<Object, Object> registry = REGISTRY.get();
179         registry.remove(element);
180     }

182     /**
183      * Called to generate the key. Depending on the {@link #setIncludeMethod(boolean)} and 
184      * {@link #setIncludeParameterTypes(boolean)} the the appropriate parts of the {@link Method}
185      * signature will be included in the data array.
186      * 
187      *  @return The {@link Serializable} cache key for the method invocation.
188      */
189     public abstract T generateKey(Object... data);
